/**
 * Adds the highlight class to the filter toggle and it's parent elements.
 * @date 2024-11-19
 * @param {object} tableToHighlight
 */
function updateHighlight(tableToHighlight) {
    while (tableToHighlight.flyout) {
        query('.crashFilterFlyoutHeader,.crashFilterFlyoutSubHeader', tableToHighlight.flyout.id).forEach(function (
            node
        ) {
            domStyle.set(node, greenButtonEffect);
            domClass.add(node, 'crashFilterSelected');
        });
        domStyle.set(tableToHighlight.toggleIcon || tableToHighlight.toggleText, greenButtonEffect);
        domClass.add(tableToHighlight.toggleIcon || tableToHighlight.toggleText, 'crashFilterSelected');
        tableToHighlight = tableToHighlight.parent;
    }
}

/**
 * Validates if a given number falls within a range or matches a single value defined in a string.
 *
 * @function checkRange
 * @description Determines whether an input number is within a specified range (e.g., "10-20") or matches a single value (e.g., "15/").
 * @date 2024-11-22
 * @param {string} inputStr - A string defining a range (e.g., "10-20") or a single value (e.g., "15/").
 * @param {number} inputNum - The number to check against the range or value.
 * @returns {boolean} `true` if the number is within the range or matches the value; otherwise, `false`.
 * @example
 * // Example 1: Check within a range
 * checkRange("10-20", 15); // Returns true
 * 
 * // Example 2: Check for an exact value
 * checkRange("15/", 15); // Returns true
 * 
 * // Example 3: Out of range
 * checkRange("10-20", 25); // Returns false
 */
function checkRange(inputStr, inputNum) {
    // Check if the inputStr contains a range or a single value
    if (!isNaN(parseInt(inputNum)) && isFinite(parseInt(inputNum))) {
        inputNum = parseInt(inputNum);
        if (inputStr.includes('-')) {
            // Parse the string to extract the range
            const [start, end] = inputStr.split('-').map(Number);

            // Check if the input number is within the range
            if (inputNum >= start && inputNum <= end) {
                return true;
            } else {
                return false;
            }
        } else if (inputStr.includes('/')) {
            // Parse the string to extract the single value
            const value = Number(inputStr.split('/')[0]);

            // Check if the input number matches the single value
            if (inputNum === value) {
                return true;
            } else {
                return false;
            }
        }         
    }

    return false;
}


/**
 * Creates a checkbox filter element in the filter menu.
 * @date 2024-11-19
 * @module FilterMenu/widgets/CheckboxFilter
 * @param {domNode} toggleAttachPoint The div where the Filter Menu button will be attached to.
 * @param {domNode} filterContents The div where the filter options will be attached to.
 * @param {object} filterInfo The object containing the filter categories and their options.
 * @param {domNode} parent The parent div of this filter toggle div.
 * @param {object} parentNode The parent object of this menu item.
 * @param {module} crashFilter The Crash Filter module.
 * @param {domNode} toggleButton The div containing the toggle button to open the filter.
 * @param {module} crashMap The Crash Map module.
 * @param {module} resultsPane The Results Pane module.
 * @requires ./widgets/CheckBoxRow
 * @requires ./widgets/ClearMenuFiltersRow
 * @requires ./checkbox.filter.css
 */
export default function CheckboxFilter(
    toggleAttachPoint,
    filterContents,
    filterInfo,
    parent,
    parentNode,
    crashFilter,
    toggleButton,
    crashMap,
    resultsPane,
) {
    const self = this;
    this.checkBoxes = [];
    this.parentNode = parentNode;
    const checkboxHeader = document.createElement('div');
    const checkboxTable = document.createElement('div');
    this.toggle = document.createElement('div');
    this.content = document.createElement('div');

    if (filterInfo.flyout.hasOwnProperty('tooltip')) {
        this.toggle.title = filterInfo.flyout.tooltip;
        if (filterInfo.label) {
            this.toggle.title += ` This filter corresponds to box number ${filterInfo.label.njtr1Number} in the NJTR-1 form.`
        }
    }
    else if (filterInfo.flyout.hasOwnProperty('title')) {
        this.toggle.title = filterInfo.flyout.title;
    }

    // this.toggle.title = filterInfo.flyout.title;
    this.toggle.id = filterInfo.flyout.id + '_toggle';

    this.content.className = 'filter-container omni-element has-title hidden';
    if (filterInfo.flyout.hasOwnProperty('id')) this.content.id = filterInfo.flyout.id; // mainly used for jurisdiction filter

    checkboxHeader.className = 'omni-title';
    checkboxHeader.innerHTML = filterInfo.flyout.title;
    checkboxTable.className = 'checkbox-table';

    this.content.append(checkboxHeader, checkboxTable);
    filterContents.appendChild(this.content);
    toggleAttachPoint.append(this.toggle);

    this.toggle.className = 'omni-element';

    if (filterInfo.toggleText) {
        if (filterInfo.label) {
            this.toggle.innerHTML = `${filterInfo.flyout.title}<span class="toggle-label"><div class="spacer"></div><span class="njtr1-number">${filterInfo.label.njtr1Number}</span></span>`;
        } else {
            this.toggle.innerHTML = `${filterInfo.flyout.title}`;
        }
    } else {
        if (filterInfo.label) {
            this.toggle.innerHTML = `<span class="toggle-label">${filterInfo.label.text}<div class="spacer"></div><span class="njtr1-number">${filterInfo.label.njtr1Number}</span></span>`;
        } else if (filterInfo.flyout.title) {
            this.toggle.innerHTML = `<span class="toggle-label">${filterInfo.flyout.title}</span>`;
        } else {
            this.toggle.innerHTML = '';
        }
    }

    if (filterInfo.toggleIcon) {
        let icon = document.createElement('div');
        icon.className = `image-icon ${filterInfo.toggleIcon}`;

        this.toggle.prepend(icon);
    } else if (filterInfo.toggleText) {
        this.toggle.classList.add('toggle-text');
    } else {
        this.toggle.classList.add('ri-filter-2-fill');
    }

    if (filterInfo.hasOwnProperty('isQuickFilter')) {
        this.isQuickFilter = filterInfo.isQuickFilter;
    } else {
        this.isQuickFilter = false;
    }

    if (filterInfo.selectAll) {
        var selectAllRow = domConstruct.create('tr', { className: '' }, this.domNode);
        this.selectAllButton = domConstruct.create(
            'td',
            {
                innerHTML: 'Select Entire County',
                className: 'deleteButton',
                colspan: 2,
                onclick: function () {
                    var currentSelectAllButton = this;
                    currentSelectAllButton.classList.add('working');
                    self.selectEntireTable(false, this);
                }
            },
            selectAllRow
        );
    }

    for (let i = 0; i < filterInfo.values.length; i++) {
        const value = filterInfo.values[i];
        this.checkBoxes.push(
            new CheckBoxRow(this, checkboxTable, value, filterInfo, crashFilter, crashMap, resultsPane)
        );
    }

    if (filterInfo.clearAll) {
        new ClearMenuFiltersRow(this, checkboxTable, crashMap, resultsPane);
    }

    this.hide = function (event) {
        self.content.classList.add('hidden');
    };
    this.show = function (event) {
        self.content.classList.remove('hidden');

    /**
     * Gets the ARD column codes and value codes from the selected filters.
     * Used to query the database.
     * @date 2022-02-07
     * @function getSelections
     * @returns {object} - {crash_type: 2,3,4, mun_mu: '01'...}
     */
    this.getSelections = function () {
        const selectionsWithCodes = [];
        const selectionsWithFields = {};
        const filter = {};
        var secondaryFieldName; // contains mun_cty_co, etc.
        self.checkBoxes.forEach((checkbox) => {
            if (checkbox.checked) {
                if (checkbox.rowAttributes.customValue && checkbox.rowAttributes.fieldName) {
                    selectionsWithFields[checkbox.rowAttributes.fieldName] = checkbox.rowAttributes.customValue;
                } else if (checkbox.rowAttributes.code && checkbox.rowAttributes.fieldName) {
                    if (filter.hasOwnProperty(checkbox.rowAttributes.fieldName)) {
                        filter[checkbox.rowAttributes.fieldName] += ',' + checkbox.rowAttributes.code;
                    } else {
                        filter[checkbox.rowAttributes.fieldName] = checkbox.rowAttributes.code;
                        secondaryFieldName = checkbox.rowAttributes.fieldName;
                    }
                    selectionsWithCodes.push(checkbox.rowAttributes.code);
                } else if (checkbox.rowAttributes.code) {
                    selectionsWithCodes.push(checkbox.rowAttributes.code);
                } else {
                    selectionsWithFields[checkbox.rowAttributes.fieldName] = '0';
                }
            }
        });

        if (selectionsWithCodes.length > 0 || Object.keys(selectionsWithFields).length > 0) {
            if (secondaryFieldName) {
                filter[secondaryFieldName] = selectionsWithCodes.join(',');
            } else {
                if (filterInfo.fieldName === 'mun_mu') {
                    var ctyCodes = [];
                    var muniCodes = [];
                    selectionsWithCodes.forEach((code) => {
                        if (!ctyCodes.includes(code.substring(0, 2))) ctyCodes.push(code.substring(0, 2));
                        muniCodes.push(code.substring(2));
                    });
                    //filter['mun_cty_co'] = ctyCodes.join(',');
                    filter[filterInfo.fieldName] = selectionsWithCodes.join(',');
                } else filter[filterInfo.fieldName] = selectionsWithCodes.join(',');
            }
        }

        Object.assign(filter, selectionsWithFields);
        return filter;
    };

    /**
     * Gets the human readable ARD column names and values from the selected filters.
     * Used to display current filters in the Results Pane.
     * @date 2022-02-07
     * @function getDescriptionSelections
     * @returns {object} - {Crash Type: 'Backing, Animal', County: 'Atlantic', ...}
     */
    this.getDescriptionSelections = function () {
        const selectionsWithCodes = [];
        const selectionsWithFields = {};
        const filter = {};
        var secondaryFieldName; // contains mun_cty_co, etc.
        self.checkBoxes.forEach((checkbox) => {
            if (checkbox.checked) {
                if (checkbox.rowAttributes.customValue && checkbox.rowAttributes.fieldName) {
                    // special category: pedestrian_phys_cond, occupant_phys_cond, trf_ctrl, driver_phys
                    selectionsWithFields[checkbox.rowAttributes.fieldName] = checkbox.rowAttributes.description;
                } else if (checkbox.rowAttributes.code && checkbox.rowAttributes.fieldName) {
                    if (filter.hasOwnProperty(checkbox.rowAttributes.fieldName)) {
                        filter[checkbox.rowAttributes.fieldName] += ', ' + checkbox.rowAttributes.description;
                    } else {
                        selectionsWithCodes.push(checkbox.rowAttributes.description);
                        secondaryFieldName = checkbox.rowAttributes.fieldName;
                    }
                } else if (checkbox.rowAttributes.code) {
                    selectionsWithCodes.push(checkbox.rowAttributes.description);
                } else {
                    // special category: pedestrian_phys_cond, occupant_phys_cond, trf_ctrl, driver_phys
                    selectionsWithFields[checkbox.rowAttributes.fieldName] = checkbox.rowAttributes.description;
                }
            }
        });

        if (selectionsWithCodes.length > 0 || Object.keys(selectionsWithFields).length > 0) {
            if (secondaryFieldName) {
                if (secondaryFieldName == 'mun_cty_co') filter['Jurisdiction'] = selectionsWithCodes.join(', ');
                else filter[secondaryFieldName] = selectionsWithCodes.join(', ');
            } else {
                if (filterInfo.fieldName === 'mun_mu') {
                    if (selectionsWithCodes.length > 3) filter['Jurisdiction'] = `(${selectionsWithCodes.length})`;
                    else filter['Jurisdiction'] = selectionsWithCodes.join(', ');
                } else if (filterInfo.fieldName === 'physcl_cndtn_code') {
                    if (selectionsWithCodes.length > 3)
                        filter[filterInfo.flyout.title + ' Physical Condition'] = `(${selectionsWithCodes.length})`;
                    filter[filterInfo.flyout.title + ' Physical Condition'] =
                        Object.values(selectionsWithFields).join(', ');
                } else if (Object.keys(selectionsWithFields).length > 0) {
                    // special category: pedestrian_phys_cond, occupant_phys_cond, trf_ctrl, driver_phys
                    filter[filterInfo.flyout.title] = Object.values(selectionsWithFields).join(', ');
                } else {
                    if (selectionsWithCodes.length > 3)
                        filter[filterInfo.flyout.title] = `(${selectionsWithCodes.length})`;
                    else filter[filterInfo.flyout.title] = selectionsWithCodes.join(', ');
                }
            }
        }

        //Object.assign(filter, selectionsWithFields);
        return filter;
    };

    /**
     * Sets the checkbox filter for each filter code in `filterValueString` by checking off the checkboxes.
     * @date 2024-11-19
     * @function setFilter
     * @param {string} filterValueString Comma seperated filter code values.
     * @example
     * checkboxFilter.checkFilterField('road_surf_code');
     * che
    this.setFilter = function (filterValueString) {
        const parsedValues = filterValueString.split(',');
        parsedValues.forEach((value) => {
            for (let index = 0; index < self.checkBoxes.length; index++) {
                const checkbox = self.checkBoxes[index];
                if (checkbox.rowAttributes.code === value) {
                    checkbox.check();
                }
            }
        });
    };

    this.searchTitles= function(searchText) {
        let nodeResults = [];
        if (
            (filterInfo.flyout.title.toUpperCase().indexOf(searchText.toUpperCase()) > -1) || 
            (filterInfo.label !== undefined && 
            (checkRange(filterInfo.label.njtr1Number, searchText) || filterInfo.label.njtr1Number === searchText))
        ) 
        // if (
        //     (filterInfo.flyout.title.toUpperCase().indexOf(searchText.toUpperCase()) > -1) || (filterInfo.label !== undefined && filterInfo.label.njtr1Number === searchText)
        // ) 
        
        {
            if (filterInfo.hasOwnProperty('toggleNode')) {
                const parentNode = document.getElementById(filterInfo['toggleNode']);
                if (parentNode) {
                    if (parentNode.hasOwnProperty('content')) {
                        if (parentNode.content.id !== 'filtersDiv' || parentNode.content.id !== 'filtersMenuDiv') nodeResults.push(parentNode);
                    }
                    else if (parentNode.hasOwnProperty('id') || parentNode.id) {
                        if (parentNode.id !== 'filtersDiv' || parentNode.id !== 'filtersMenuDiv') nodeResults.push(parentNode);
                    }
                }
            }

            if (this.id !== 'filtersDiv' || this.id !== 'filtersMenuDiv') nodeResults.push(this.content);
            if (this.hasOwnProperty('toggle') && (this.id !== 'filtersDiv' || this.id !== 'filtersMenuDiv')) nodeResults.push(this.toggle);
        }
        return nodeResults;
    }

    this.searchCheckboxes = function(searchText) {
        let nodeResults = [];
        for (let index = 0; index < self.checkBoxes.length; index++) {
            const checkbox = self.checkBoxes[index];

            if (
                (checkbox.rowAttributes.description && checkbox.rowAttributes.description.toUpperCase().includes(searchText.toUpperCase())) ||
                (checkbox.rowAttributes.code !== undefined && checkbox.rowAttributes.code === searchText)
            ) {
                nodeResults.push(checkbox.description);
                // add the parent toggle node to be highlighted
                if (checkbox.parentToggle.id !== 'filtersDiv' || parentNode.id !== 'filtersMenuDiv') {
                    nodeResults.push(checkbox.parentToggle);
                }
                // nodeResults.push(self.parentNode.toggle);

                if (self.hasOwnProperty('parentNode')) {
                    nodeResults.push(self.parentNode.toggle);
                }
            }
        }
        return nodeResults;
    }

    this.searchFilter = function(searchText) {
        let results = [];
        const titleResults = self.searchTitles(searchText);
        const checkboxResults = self.searchCheckboxes(searchText);
        results.push(...titleResults, ...checkboxResults);
        return results;
    }


    /**

                checkbox.uncheck();
            }
            checkbox.disable(false);
        });
        self.content.classList.remove('selected');
        self.toggle.classList.remove('green-button-toggle');
    };

    /**
     * Highlights the toggle button if there are selected checkboxes. 
     * If there are no checked items, disables the highlight for the toggle and filter sidebar button.
     * If the jurisdiction filters are selected, includes the descriptions to the Omni Search input box.
     * @date 2024-11-19

        self.toggle.classList.remove('green-button-toggle');
        self.toggle.classList.add('orange-button-toggle');
        self.content.classList.remove('selected');

        for (let index = 0; index < self.checkBoxes.length; index++) {
            const checkbox = self.checkBoxes[index];
            if (checkbox.checked) {
                self.toggle.classList.add('green-button-toggle');
                self.toggle.classList.remove('orange-button-toggle');
                self.content.classList.add('selected');
            }
        }

        toggleButton.highlight(); // highlight the parent filter button

        // process the juri selections if there are items checked
        const juriSelections = crashFilter.getSelectedJurisdictions();
        if (juriSelections.codes.length > 0) {
            // crashFilter.setOmniSearchWithSelections(juriSelections);    // set the Omni Search input box with jurisdiction selections
            crashFilter.disableJurisdictionItems(juriSelections); //
            // crashMap.zoomToJurisdictionSelections(juriSelections);  // zoom to the bounding box of the selected jurisdictions
        } else {
            crashFilter.enableJurisdictionItems();
        }
    };

    this.toggle.addEventListener('click', function (e) {
        const parentSidebar = this.parentNode;
        const filterContainers = document.querySelectorAll('.filter-container');
        const menuToggles = parentSidebar.querySelectorAll('.omni-element');

        if (self.content.classList.contains('hidden')) {

                filterContainer.classList.add('hidden');
            });

            menuToggles.forEach((toggle) => {
                if (toggle !== this) {
                    toggle.classList.remove('orange-button-toggle');
                }
            });

            Array.prototype.forEach.call(parentSidebar.parentNode.children, function (child) {
                if (
                    child !== parentSidebar &&
                    child !== parent.content &&
                    child.classList.contains('sidebar-container')
                ) {
                    child.classList.add('hidden');
                }
            });

            self.show();
        } else {
            self.hide();
        }
    });

    this.toggle.addEventListener('mouseenter', function (event) {
        if (!this.classList.contains('green-button-toggle')) {
            this.classList.add('orange-button-toggle');
        }
    });

        if (self.content && self.content.classList.contains('hidden'))
            self.toggle.classList.remove('orange-button-toggle');

    /**
     * Clears all checkbox filters within a category.
     * @date 2024-11-19
     * @function clearTable
     * @param {boolean} global If false, updates the graphics in the map interface.
     * @param {domNode} clearTableButton The button div enabling the user to clear all checkbox filters.
     */
    this.clearTable = function (global, clearTableButton) {
        var flyoutInfo = filterInfo;
        query('#' + flyoutInfo.flyout.id + ' .filter-description').forEach(function (node) {
            domStyle.set(node, whiteButtonEffect);
            domClass.remove(node, 'crash-filter-selected');
            domClass.remove(node, 'disabled');
        });
        query('#' + flyoutInfo.flyout.id + ' .filter-check-button ').forEach(function (node) {
            domStyle.set(node, whiteButtonEffect);
            domClass.remove(node, 'crash-filter-selected');
            domClass.remove(node, 'disabled');
        });
        while (flyoutInfo.flyout) {
            if (query('.filter-check-button.crash-filter-selected', flyoutInfo.flyout.id).length > 0) {
            } else if (query('.filterTextButton.crash-filter-selected', flyoutInfo.flyout.id).length > 0) {
            } else {
                domStyle.set(
                    flyoutInfo.toggleIcon || flyoutInfo.toggleText,
                    global ? whiteButtonEffect : orangeButtonEffect
                );
                domClass.remove(flyoutInfo.toggleIcon || flyoutInfo.toggleText, 'crash-filter-selected');
                query('.crashFilterFlyoutHeader,.crashFilterFlyoutSubHeader', flyoutInfo.flyout.id).forEach(function (
                    node
                ) {
                    domStyle.set(node, orangeButtonEffect);
                });
            }
            flyoutInfo = flyoutInfo.parent;
        }
        if (global) {
        } else {
            var filter = new DataFilter();
            if (filter.content.SRI) {
                if ((filter.content.SRI && filter.content.mp_start) || filter.content.mp_end) {
                    var definitionString = "SRI = '" + filter.content.SRI + "'";
                    var routeDetails = { SRI: filter.content.SRI };
                    if (filter.content.mp_start) {
                        definitionString += ' AND MP >= ' + filter.content.mp_start;
                        routeDetails['mp_start'] = filter.content.mp_start;
                    }
                    if (filter.content.mp_end) {
                        definitionString += ' AND MP <= ' + filter.content.mp_end;
                        routeDetails['mp_end'] = filter.content.mp_end;
                    }
                    sriOverlayLayer.updateClientSideGraphics(
                        {
                            ResultType: 'SRI',
                            ResultID: filter.content.SRI,
                            ResultText: '',
                            definitionExpression: definitionString,
                            sriWithMP: routeDetails
                        },
                        clearTableButton,
                        searchOmniBox.locationFilter,
                        false,
                        false
                    );
                } else {
                    sriOverlayLayer.updateClientSideGraphics(
                        {
                            ResultType: 'SRI',
                            ResultID: filter.content.SRI,
                            ResultText: ''
                        },
                        clearTableButton,
                        searchOmniBox.locationFilter,
                        false,
                        false
                    );
                }
            } else if (self.zoomElement && self.zoomElement === true) {
                updateMapGraphics(clearTableButton, rowAttributes.zoomElement, this);
            } else {
                updateMapGraphics(clearTableButton, false, 'createCheckBoxRow');
            }
        }
    };

     * Selects all checkbox filters within a category.

     * @function selectEntireTable

            domStyle.set(node, green - button - effect);
            domClass.add(node, 'crash-filter-selected');
        });
        updateHighlight(flyoutInfo);
        if (global) {
        } else {
            updateMapGraphics(null, false, 'selectAllButton');
        }
        selectAllButton.classList.remove('working');
    };
