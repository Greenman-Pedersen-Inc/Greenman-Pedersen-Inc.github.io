import './filter.menu.css';

import DateSelector from './widgets/DateSelector';
import FilterReset from './widgets/FilterReset';
import LocationFilter from './widgets/LocationFilter';
import MultivehicleFilter from './widgets/MultivehicleFilter';
import CheckboxFilter from './widgets/CheckboxFilter';
import ExtentsReset from './widgets/ExtentsReset';
import FilterSearch from './widgets/FilterSearch';

/**
 * Populates the filter contents div with filter categories from the `filterTree` object. Creates the first tier of flyouts that link to the filter buttons.
 * @date 2024-11-18
 * @param {domNode} toggleAttachPoint The div where this filter toggle div will be appended to.
 * @param {domNode} filterContents The div where the filter options will be attached to.
 * @param {object} filterTree The object containing the filter categories and their options.
 * @param {domNode} parent The parent div of this filter toggle div.
 * @param {object} parentNode The parent object of this menu item.
 * @param {module} crashFilter The Crash Filter module.
 * @param {domNode} toggleButton The div containing the toggle button to open the filter.
 * @param {module} crashMap The Crash Map module.
 * @param {module} resultsPane The Results Pane module.
 * @param {module} statsPane The Statistics Pane module.
 * @returns {module} The Crash Filter module.
 */
function createWidgetContent(
    toggleAttachPoint,
    filterContents,
    filterTree,
    parent,
    parentNode,
    crashFilter,
    toggleButton,
    crashMap,
    resultsPane,
    statsPane
) {
    // createWidgetContent: creates the filter list along the right side.
    // also creates the first tier of flyouts that link to the filter buttons.
    // @param {object} filterContent: menu items in the filter flyout

    if (filterTree.widgets && filterTree.widgets.length > 0) {
        let widgetRoot;
        for (let i = 0; i < filterTree.widgets.length; i++) {
            const widgetInfo = filterTree.widgets[i];

            if (widgetInfo.type === 'date') {
                widgetRoot = document.querySelectorAll('#' + filterTree.flyout.id + ' .crashFilterFlyoutContent')[0];
                const dateSelector = new DateSelector(
                    parent,
                    crashMap,
                    resultsPane,
                    toggleAttachPoint,
                    filterContents,
                    filterTree,
                    widgetInfo.startYear,
                    widgetInfo.endYear,
                    widgetInfo.defaultStartYear,
                    widgetInfo.defaultEndYear,
                    toggleButton
                );
                crashFilter.filterWidgets.push(dateSelector);
                crashFilter.filterWidgetToggles.push(dateSelector.toggle);
            } else if (widgetInfo.type === 'location') {
                widgetRoot = document.querySelectorAll('#' + filterTree.flyout.id + ' .crashFilterFlyoutContent')[0];
                const locationFilter = new LocationFilter(
                    parent,
                    crashFilter,
                    toggleAttachPoint,
                    filterContents,
                    filterTree,
                    toggleButton,
                    crashMap,
                    resultsPane
                );
                crashFilter.filterWidgets.push(locationFilter);
                crashFilter.filterWidgetToggles.push(locationFilter.toggle);
            } else if (widgetInfo.type === 'multivehicle') {
                widgetRoot = document.querySelectorAll('#' + filterTree.flyout.id + ' .crashFilterFlyoutContent')[0];
                const multivehicleFilter = new MultivehicleFilter(
                    parent,
                    crashFilter,
                    toggleAttachPoint,
                    filterContents,
                    filterTree,
                    toggleButton,
                    crashMap,
                    resultsPane
                );
                crashFilter.filterWidgets.push(multivehicleFilter);
                crashFilter.filterWidgetToggles.push(multivehicleFilter.toggle);
            } else if (widgetInfo.type === 'clearAllFilters') {
                crashFilter.resetWidget = new FilterReset(
                    crashMap,
                    crashFilter,
                    resultsPane,
                    toggleAttachPoint,
                    filterTree,
                    statsPane
                );
            } else if (widgetInfo.type === 'zoomExtents') {
                crashFilter.zoomExtents = new ExtentsReset(crashMap, toggleAttachPoint);
            }
        }
    }

    if (filterTree.values && filterTree.values.length > 0) {
        const checkboxFilter = new CheckboxFilter(
            toggleAttachPoint,
            filterContents,
            filterTree,
            parent,
            parentNode,
            crashFilter,
            toggleButton,
            crashMap,
            resultsPane
        );
        crashFilter.filterWidgets.push(checkboxFilter);
        if (!checkboxFilter.toggle.classList.contains('toggle-text')) {
            crashFilter.filterWidgetToggles.push(checkboxFilter.toggle);
        }
    } else {
    } // No checkboxes were specified for this widget.

    return crashFilter;
}

/**
 * Populates a filter cateogry with values as checkbox rows.
 * @date 2024-11-18
 * @param {domNode} toggleAttachPoint The div where this filter toggle div will be appended to.
 * @param {domNode} filterToggles The div where the filter category buttons will be attached to.
 * @param {domNode} filterContents The div where the filter options will be attached to.
 * @param {object} filterTree The object containing the filter categories and their options.
 * @param {domNode} parent The parent div of this filter toggle div.
 * @param {module} crashFilter The Crash Filter module.
 * @param {domNode} toggleButton The div containing the toggle button to open the filter.
 * @param {module} crashMap The Crash Map module.
 * @param {module} resultsPane The Results Pane module.
 * @returns {module} The Crash Filter module.
 */
function populateFilterMenu(
    toggleAttachPoint,
    filterToggles,
    filterContents,
    filterTree,
    parent,
    parentNode,
    crashFilter,
    toggleButton,
    crashMap,
    resultsPane,
    statsPane
) {
    if (filterTree.menus && filterTree.menus.length > 0) {
        for (let i = 0; i < filterTree.menus.length; i++) {
            const menu = filterTree.menus[i];

            crashFilter = createWidgetContent(
                toggleAttachPoint,
                filterContents,
                menu,
                parent,
                parentNode,
                crashFilter,
                toggleButton,
                crashMap,
                resultsPane,
                statsPane
            );

            // if there are nested menus, push values to a new submenu
            if (menu && menu.menus) {
                let filterMenuItem = new FilterMenu(
                    toggleAttachPoint,
                    filterToggles,
                    filterContents,
                    menu,
                    parent,
                    parentNode,
                    crashFilter,
                    crashMap,
                    resultsPane
                );

                crashFilter = filterMenuItem.crashFilter;
                crashFilter.filterWidgetToggles.push(filterMenuItem.toggle);
            }
        }
    }

    return crashFilter;
}

/**
 * Creates the entire filter menu system with all filter categories and values.
 * @date 2024-11-18
 * @module FilterMenu
 * @param {domNode} toggleAttachPoint The div where the Filter Menu button will be attached to.
 * @param {domNode} filterToggles The div where the filter category buttons will be attached to.
 * @param {domNode} filterContents The div where the filter options will be attached to.
 * @param {object} filterTree The object containing the filter categories and their options.
 * @param {boolean|object} parent=false If false, this Filter Menu is the overarching menu. If the `parent` is a DOM node, then this filter menu is a category.
 * @param {object} parentNode The parent object of this menu item.
 * @param {module} crashFilter The Crash Filter module.
 * @param {module} crashMap The Crash Map module.
 * @param {module} resultsPane The Results Pane module.
 * @requires ./filter.menu.css
 * @requires ./widgets/DateSelector
 * @requires ./widgets/FilterReset
 * @requires ./widgets/LocationFilter
 * @requires ./widgets/MultivehicleFilter
 * @requires ./widgets/CheckboxFilter
 * @requires ./widgets/ExtentsReset
 * @requires ./widgets/SignalsFilter
 */
export default function FilterMenu(
    toggleAttachPoint,
    filterToggles,
    filterContents,
    filterTree,
    parent = false,
    parentNode,
    crashFilter,
    crashMap,
    resultsPane,
    statsPane,
    id = undefined,
    initFilterSearch = false
) {
    const self = this;
    // this is will fire when the window resizes, when the menu opens, and when the results pane is toggled
    this.crashFilter = crashFilter;
    this.toggle = document.createElement('div');
    this.toggle.className = 'omni-element';

    if (filterTree.toggleText) {
        this.toggle.innerHTML = filterTree.flyout.title;
    } else {
        if (filterTree.label) {
            this.toggle.innerHTML = `<span class="toggle-label">${filterTree.label.text}<div class="spacer"></div><span class="njtr1-number">${filterTree.label.njtr1Number}</span></span>`;
        } else if (filterTree.flyout.title) {
            this.toggle.innerHTML = `<span class="toggle-label">${filterTree.flyout.title}</span>`;
        } else {
            this.toggle.innerHTML = '';
        }
    }

    if (filterTree.toggleIcon) {
        let icon = document.createElement('div');
        icon.className =
            `image-icon ${filterTree.toggleIcon} ` + `${filterTree.classes ? `${filterTree.classes}` : ''}`;
        this.toggle.prepend(icon);
    } else if (filterTree.toggleText) {
        this.toggle.classList.add('toggle-text');
    } else {
        this.toggle.classList.add('ri-filter-2-fill');
    }

    if (filterTree.flyout.hasOwnProperty('tooltip')) {
        this.toggle.title = filterTree.flyout.tooltip;
    } else if (filterTree.flyout.hasOwnProperty('title')) {
        this.toggle.title = filterTree.flyout.title;
    } else if (filterTree.flyout.hasOwnProperty('rootTitle')) {
        this.toggle.title = filterTree.flyout.rootTitle;
    }

    if (filterTree.label) {
        this.toggle.title += ` This filter corresponds to box number ${filterTree.label.njtr1Number} in the NJTR-1 form.`
    }

    if (filterTree.flyout.hasOwnProperty('id')) this.toggle.id = filterTree.flyout.id;

    this.content = document.createElement('div');
    this.content.className = 'sidebar-container flyout omni-menu hidden';
    if (id !== undefined) this.content.id = id;
    if (initFilterSearch) {
        const filterSearch = new FilterSearch(crashFilter);
        this.content.append(filterSearch);
    }

    if (filterTree.flyout && filterTree.flyout.title) {
        this.filterTitle = document.createElement('div');
        this.filterTitle.className = 'omni-title';
        this.filterTitle.innerHTML = filterTree.flyout.title;
        this.content.classList.add('has-title');
        this.content.append(this.filterTitle);
    }

    // Internal Attributes and Functions
    
    /**
     * Updated the filter menu with loading spinner gif if a filter was selected and the module is waiting for the server to process the filter selection.
     * @date 2024-11-18
     * @function isLoading
     * @param {boolean} isLoading If true, then displays a loading spinner in the toggle button.
     */
    this.isLoading = function (isLoading) {
        if (isLoading) {
            self.toggle.classList.remove('ri-filter-2-fill');
            self.toggle.classList.add('filter-loading');
        } else {
            self.toggle.classList.remove('filter-loading');
            self.toggle.classList.add('ri-filter-2-fill');
        }
    };

    /**
     * Clears all loading spinners in the filters menu.
     * @function clearCheckBoxSpinners
     * @date 2024-11-18
     */
    this.clearCheckBoxSpinners = function () {
        var spinners = document.getElementsByClassName('filter-check-button crash-filter-working');
        for (var i = 0; i < spinners.length; ++i) {
            var checkBox = spinners[i];
            if (checkBox.classList.contains('crash-filter-working')) checkBox.classList.remove('crash-filter-working');
            if (checkBox.classList.contains('green-button-toggle')) checkBox.classList.add('crash-filter-selected');
        }
    };

    /**
     * Hides a filter flyout div.
     * @date 2024-11-18
     * @function hide
     */
    this.hide = function (event) {
        self.content.classList.add('hidden');
    };

    /**
     * Hides all filter flyout divs.
     * @date 2024-11-18
     */ 
    this.hideAll = function () {
        // hide all items from 'filter-toggles' and 'filter-contents' divs
        for (var i = 0; i < filterToggles.children.length; i++) {
            filterToggles.children[i].classList.add('hidden');
        }
        for (var i = 0; i < filterContents.children.length; i++) {
            filterContents.children[i].classList.add('hidden');
        }

        self.toggle.highlight();
    };

    /**
     * Displays a filter flyout div.
     * @date 2024-11-18
     * @function show
     */
    this.show = function (event) {
        self.content.classList.remove('hidden');
    };

    /**
     * Adds the button highlight class to the Filter Menu toggle.
     * @date 2024-11-18
     * @function toggle.highlight
     */
    this.toggle.highlight = function () {
        self.toggle.classList.remove('green-button-toggle');

        if (self.content.classList.contains('hidden')) {
            self.toggle.classList.remove('orange-button-toggle');
        } else {
            self.toggle.classList.add('orange-button-toggle');
        }

        if (filterTree.flyout && filterTree.flyout.title) {
            self.content.classList.remove('selected');
        }

        // the child array is an HTML Element Collection which you cannot iterate over
        // you use the prototype function to call the array-like object
        Array.prototype.forEach.call(self.content.children, function (child) {
            if (child.classList.contains('green-button-toggle')) {
                self.toggle.classList.add('green-button-toggle');
                self.toggle.classList.remove('orange-button-toggle');

                if (filterTree.flyout && filterTree.flyout.title) {
                    self.content.classList.add('selected');
                }
            }
        });

        if (parent) {
            parent.toggle.highlight();
        }
    };

    // Event Listeners
    this.toggle.addEventListener('click', function (e) {
        const hidden = self.content.classList.contains('hidden');
        const parentSidebar = this.parentNode;
        const filterContainers = document.querySelectorAll('.filter-container');
        const omniElements = self.content.querySelectorAll('.omni-element');
        const menuToggles = parentSidebar.querySelectorAll('.omni-element');

        document.getElementById('settingsButton').classList.remove('green-button-toggle');
        document.getElementById('alertsButton').classList.remove('green-button-toggle');

        filterContainers.forEach((filterContainer) => {
            filterContainer.classList.add('hidden');
        });

        // this will remove the orange toggles when the menu closes
        // this accounts for the button staying orange while menu is open
        omniElements.forEach((element) => {
            element.classList.remove('orange-button-toggle');
        });

        Array.prototype.forEach.call(filterToggles.children, function (child) {
            if (child !== parentSidebar && child.classList.contains('sidebar-container')) {
                child.classList.add('hidden');
            }
        });

        if (hidden) {
            menuToggles.forEach((toggle) => {
                if (toggle !== this) {
                    toggle.classList.remove('orange-button-toggle');
                }
            });

            self.show();
        }

        // hide the alerts div
        document.querySelector('.updates-box').classList.add('hidden');
        // if there are OmniSearch results, hide them as they block the menu
        document.querySelector('.search-omni-box-results').classList.add('hidden');
    });
    this.toggle.addEventListener('mouseenter', function () {
        if (!self.toggle.classList.contains('green-button-toggle')) {
            self.toggle.classList.add('orange-button-toggle');
        }
    });
    this.toggle.addEventListener('mouseleave', function () {
        if (self.content && self.content.classList.contains('hidden'))
            self.toggle.classList.remove('orange-button-toggle');
    });

    // Construct the Menu
    toggleAttachPoint.append(this.toggle);
    filterToggles.append(this.content);

    populateFilterMenu(
        this.content,
        filterToggles,
        filterContents,
        filterTree,
        parent ? parent : self,
        self,
        this.crashFilter,
        this.toggle,
        crashMap,
        resultsPane,
        statsPane
    );

    return this;
}
