import StatsExplorer from './StatsExplorer';
import API from '../API';
import './stats.pane.css';

/**
 * Creates a button on the map UI that opens or closes the Statistics Pane.
 * @date 2024-12-16
 * @param {function} map The Mapbox map.
 * @param {domNode} containerDiv The target div that will open or close when the button is collapse clicked.
 */
function addCollapseButton(map, containerDiv) {
    /**
     * Toggles the collapse state of a statistics pane and adjusts the map container's layout.
     *
     * @param {HTMLElement} buttonDiv - The button element used to toggle the collapse state.
     * @param {HTMLElement} div - The div element representing the statistics pane to be collapsed or expanded.
     */
    function collapseDiv(buttonDiv, div) {
        const mapContainer = document.getElementById('map-container');
        const resultsPane = document.getElementById('results-pane');
        const mapResizer = setInterval(function () {
            map.resize();
        }, 10);

        if (div.classList.contains('collapse-div')) {
            div.classList.remove('collapse-div');
            resultsPane.classList.remove('full-width');
            mapContainer.classList.remove('full-width');
            buttonDiv.title = 'Collapse statistics pane';
            buttonDiv.classList.remove('ri-expand-left-line');
            buttonDiv.classList.add('ri-expand-right-line');
        } else {
            div.classList.add('collapse-div');
            resultsPane.classList.add('full-width');
            mapContainer.classList.add('full-width');
            buttonDiv.title = 'Expand statistics pane';
            buttonDiv.classList.remove('ri-expand-right-line');
            buttonDiv.classList.add('ri-expand-left-line');
        }
        map.resize();

        setTimeout(function () {
            clearInterval(mapResizer);
        }, 500);
    }

    class CollapseButton {
        onAdd(map) {
            const buttonDiv = document.createElement('div');
            buttonDiv.className = 'mapboxgl-ctrl mapboxgl-ctrl-group ri-expand-left-line omni-element mt-0';
            buttonDiv.title = 'Collapse statistics pane';
            buttonDiv.id = 'stats-pane-toggle';
            buttonDiv.addEventListener('click', () => collapseDiv(buttonDiv, containerDiv));
            return buttonDiv;
        }
    }
    const collapseButton = new CollapseButton();
    map.addControl(collapseButton, 'top-right');
}

/**
 * Creates a pill HTML element based on a list of pill objects
 * @date 3/15/2024 - 1:40:04 PM
 *
 * @param {list} pillItems list of objects with the structure: { "text": pill text, "classes": classes attached to the pill, "value": custom value to be sent to fetch request}
 * @returns {HTMLElement} pill html object
 */
function createPillMenu(pillItems, parentDiv) {
    const pillDiv = document.createElement('div');
    const pillContentDiv = document.createElement('div');
    pillContentDiv.classList = 'tab-content';

    // Create the ul element with class "nav" and "nav-pills"
    let ulElement = document.createElement('ul');
    ulElement.className = 'nav nav-pills m-3';

    let i = 0;
    for (const [key, value] of Object.entries(pillItems)) {
        // Create the li element with class "nav-item"
        let liElement = document.createElement('li');
        liElement.className = 'nav-item';
        liElement.setAttribute('role', 'presentation');

        // Create the a element with class "nav-link" and set text content, href, and additional class
        let aElement = document.createElement('button');
        if (i === 0) {
            aElement.className = 'nav-link active';
            aElement.setAttribute('aria-selected', 'true');
        } else {
            aElement.className = 'nav-link';
            aElement.setAttribute('aria-selected', 'false');
        }
        aElement.setAttribute('data-bs-toggle', 'pill');
        aElement.setAttribute('data-bs-target', '#' + value.id + '-content');
        aElement.setAttribute('aria-controls', value + '-content');
        aElement.textContent = value.label;
        aElement.setAttribute('option-value', value.id);
        aElement.id = value.id + '-pill';
        aElement.addEventListener('click', function (event) {
            if (key === 'detailed-crash-information') {
                var grid = document.getElementById('crash-grouping-table').__ag_grid;
                grid.gridOptions.api.sizeColumnsToFit();
            }
        });

        // Append the anchor element to the list item
        liElement.appendChild(aElement);

        // Append the list item to the unordered list
        ulElement.appendChild(liElement);

        // Create pill content divs
        const tabPane = document.createElement('div');
        tabPane.classList = 'tab-pane fade';
        if (i === 0) {
            tabPane.classList = 'tab-pane fade show active';
        } else {
            tabPane.classList = 'tab-pane fade';
        }
        tabPane.id = value.id + '-content';
        tabPane.setAttribute('role', 'tabpanel');
        tabPane.setAttribute('aria-labelledby', value.id + '-tab');
        pillContentDiv.appendChild(tabPane);

        value['domNode'] = tabPane;
        i++;
    }

    pillDiv.append(ulElement, pillContentDiv);
    parentDiv.appendChild(pillDiv);
    return pillItems;
}

/**
 * Creates the Statistics Pane that contains vehicle-related statistics from crash segments.
 * Statistics include vehicle attributes, persons severity counts, and vehicle make/model charts.
 * @date 2024-12-16
 * @module StatsPane
 * @param {domNode} statsContainer The div containing the Statistics Pane.
 * @param {domNode} mapContainer The div containing the Mapbox map.
 * @param {module} crashMap The Crash Map module.
 * @param {module} crashFilter The Crash Filter module.
 */
export default function StatsPane(statsContainer, mapContainer, crashMap, crashFilter) {
    const self = this;
    const api = new API();
    self.map = crashMap.map;

    Object.defineProperty(this, 'currentLayer', {
        get: function () {
            console.log(this['_currentLayer']);
            return this['_currentLayer'];
        },
        set: function (value) {
            this['_currentLayer'] = value;

            // select the first layer in the reference layers and visualize that
            if (value) {
                api.getLayerAttributes(value).then((response) => {
                    console.log(response);
                    if (attributeSelect) {
                        attributeSelect.populate(response);
                    }
                });
            } else {
                if (attributeSelect) {
                    attributeSelect.clear();
                }
            }
        },
        enumerable: false,
    });

    crashMap.StatsPane = this;

    let pillItems = {
        'detailed-crash-information': {
            label: 'Detailed Crash Information',
            id: 'detailed-crash-information',
        },
        'vehicles-explorer': {
            label: 'Vehicles Attributes Explorer',
            id: 'vehicles-explorer',
        },
    };

    // create overlay div
    let overlayDiv = document.createElement('div');
    overlayDiv.id = 'stats-container-overlay';
    let overlayText = document.createElement('div');
    overlayText.innerHTML = 'Select an SRI segment or crash cluster to load statistics';
    overlayText.classList = 'overlay-text';
    overlayDiv.append(overlayText);
    statsContainer.append(overlayDiv);

    // Create the outer div with class "card"
    let cardDiv = document.createElement('div');
    cardDiv.className = 'card rounded-0 hidden';

    // Create header for SRI and MP range
    const cardHeaderDiv = document.createElement('h5');
    cardHeaderDiv.classList = 'card-header';
    cardHeaderDiv.id = 'stats-pane-header';
    cardDiv.append(cardHeaderDiv);

    // Create pill items
    const pillItemsDiv = createPillMenu(pillItems, cardDiv);

    // Create the div with class "card-body"
    let cardBodyDiv = document.createElement('div');
    cardBodyDiv.className = 'card-body ps-1 pe-0 pt-0 pb-0';
    cardBodyDiv.id = 'crash-details-div';

    // Append cardHeaderDiv and cardBodyDiv to cardDiv
    cardDiv.appendChild(cardBodyDiv);

    // Append cardDiv to the body or any other desired parent element
    statsContainer.appendChild(cardDiv);

    // Container element for the Crash Details pane
    const container = document.createElement('div');
    container.classList.add('container', 'text-center');

    // Creating row element
    const row = document.createElement('div');
    row.classList.add('row');

    // Appending row to the container
    container.appendChild(row);
    cardBodyDiv.appendChild(container);

    const statsExplorer = new StatsExplorer(crashMap, crashFilter);
    const vehiclesDiv = pillItemsDiv['vehicles-explorer']['domNode'];
    vehiclesDiv.appendChild(statsExplorer.domNode);

    crashMap.map.on('load', function () {
        addCollapseButton(crashMap.map, statsContainer);
    });

    /**
     * Initiates the Statistics Pane by rendering specific elements.
     * @date 2024-12-16
     * @function init
     */
    this.init = function () {
        // topLocationsTable.update();
        statsExplorer.init();
        self.enableOverlay(true);
    };

    /**
     * Updates each element in the Statistics Pane.
     * @date 2024-12-16
     * @param {object} filters An object containing crash cluster or crash segment data.
     * @function update 
     */
    this.update = function (filters) {
        statsExplorer.update(filters);
        if (filters.hasOwnProperty('sri'))  cardHeaderDiv.innerHTML = `SRI: ${filters['sri']} (Milepost ${filters['mp_start']}-${filters['mp_end']})`;
        else cardHeaderDiv.innerHTML = `Crash Cluster Details`;
        if (!self.checkOpen()) self.togglePane(true);
        self.enableOverlay(false);

    };

    /**
     * Gets the tab div for the Detailed Crash Information.
     * @date 2024-12-16
     * @returns {domNode} The tab div for the Detailed Crash Information.
     */
    this.getCrashDiv = function () {
        return document.getElementById('detailed-crash-information-content');
    };

    /**
     * Displays the overlay div over the Statistics Pane when a crash segment needs to be selected to display statistics.
     * @date 2024-12-16
     * @param {boolean} enable=true If true, displays an overlay div in the Statistics Pane that states 'Select an SRI segment or crash cluster to load statistics'.
     */
    this.enableOverlay = function (enable = true) {
        if (enable) {
            overlayDiv.classList.remove('hidden');
            cardDiv.classList.add('hidden');
        } else {
            overlayDiv.classList.add('hidden');
            cardDiv.classList.remove('hidden');
        }
    };

    /**
     * Returns a boolean based on the display status of the Statistics Pane. If it's open, returns `true`.
     * @date 2024-12-16
     * @function checkOpen
     * @returns {boolean} Returns `true` if the Statistics Pane is open; `false` if the pane is closed.
     */
    this.checkOpen = function () {
        if (statsContainer.classList.contains('collapse-div')) return false;
        else return true;
    };

    /**
     * Toggles the visibility of the statistics pane and adjusts the map container layout.
     * @date 2024-12-16
     * @function togglePane
     * @param {boolean} open=true If true, 
     */
    this.togglePane = function (open = true) {
        const resultsPane = document.getElementById('results-pane');
        const buttonDiv = document.getElementById('stats-pane-toggle');
        
        const mapResizer = setInterval(function () {
            crashMap.map.resize();
        }, 10);

        if (open) {
            statsContainer.classList.remove('collapse-div');
            resultsPane.classList.remove('full-width');
            mapContainer.classList.remove('full-width');
            buttonDiv.title = 'Collapse statistics pane';
            buttonDiv.classList.remove('ri-expand-left-line');
            buttonDiv.classList.add('ri-expand-right-line');
        } else {
            statsContainer.classList.add('collapse-div');
            resultsPane.classList.add('full-width');
            mapContainer.classList.add('full-width');
            buttonDiv.title = 'Expand statistics pane';
            buttonDiv.classList.remove('ri-expand-right-line');
            buttonDiv.classList.add('ri-expand-left-line');
        }
        crashMap.map.resize();

        setTimeout(function () {
            clearInterval(mapResizer);
        }, 500);
    };
}
