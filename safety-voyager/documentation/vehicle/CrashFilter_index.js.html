/**
 * Handles getters and setters when the user selects the crash filters.
 * The crashMap object is added to this from CrashMap function map.on('load). This allows the crashMap to update the heatmap layers when the crashFilter is updated.
 * @date 2024-10-30
 * @module CrashFilter
 */
export default function CrashFilter() {
    const self = this;
    this.highlightClassName = 'filter-search-highlight';
    this.filterWidgets = []; // holds all of the filter widgets and their respective checkbox table and row data. use this to search for selected filters or codes
    this.filterWidgetToggles = []; // holds all of the filter widgets and their respective checkbox table and row data. use this to search for selected filters or codes
    this.resetWidget;
    this.crashMap;

    /**
     * Gets all select filters from the FilterMenu widgets.
     * @date 2022-02-07
     * @function getSelectedFilters
     * @returns {object} selectedFilters - key are ard column names; values are ARD codes
     */
    this.getSelectedFilters = function(selectedFilters = {}) {
        self.filterWidgets.forEach((widget) => {
            const filters = widget.getSelections();

            // loop through the selected filters in the widget and append them to the selectedFilters object. 
            // this is so multiple counties and muni values can be queried instead of being overwritten when using the Object.assign method.
            for (const [key, value] of Object.entries(filters)) {
                if (selectedFilters.hasOwnProperty(key)) {
                    selectedFilters[key] += ',' + value;
                } else {
                    selectedFilters[key] = value;
                    // console.log(key, value);
                }
            }
        });

        return selectedFilters;
    }

    /**
     * Gets all select filters from the signal widgets.
     * @date 2022-06-21
     * @function getSignalsSelectedFilters
     * @returns {object} selectedFilters - key are ard column names; values are ard codes
     */
    this.getSelectedFiltersDescriptions = function() {
        const selectedFiltersDescriptions = {};
        self.filterWidgets.forEach((widget) => {
            const descriptionFilters = widget.getDescriptionSelections();

            // loop through the selected filters in the widget and append them to the selectedFilters object. 
            // this is so multiple counties and muni values can be queried instead of being overwritten when using the Object.assign method.
            for (const [key, value] of Object.entries(descriptionFilters)) {
                if (selectedFiltersDescriptions.hasOwnProperty(key)) {
                    selectedFiltersDescriptions[key] += ',' + value;
                } else {
                    selectedFiltersDescriptions[key] = value;
                }
            }
        });
        return selectedFiltersDescriptions;
    }

    /**
     * Takes a filter object and sets the filter menu with selections; updates map with filter selections.
     * @date 2022-03-15
     * @function setFilters
     * @param {object} crashFilterObject The currently selected crash filters. 
     * @param {boolean} isQuickFilterMenu True if the filter menu is using the quick filter structure.
     * @example
     * const crashFilterObject = { year: "2017,2018", mun_mu: "0102", crash_type: ... };
     * crashFilter.setFilters(crashFilterObject, false);
     */
    this.setFilters = function (crashFilterObject, isQuickFilterMenu) {
        // for each filter attribute, search through the widgets list for the correct widget based on fieldName.
        for (const [fieldName, selectionString] of Object.entries(crashFilterObject)) {   
            for (let index = 0; index < self.filterWidgets.length; index++) {
                const widget = self.filterWidgets[index];
                const targetWidget = widget.checkFilterField(fieldName);
                if (targetWidget !== undefined) {
                    if (fieldName === 'sri') {
                        targetWidget.setFilter(crashFilterObject.sri, crashFilterObject.mp_start, crashFilterObject.mp_end);
                        targetWidget.disableToggleButton(false);
                    }
                    else if (fieldName === 'tot_veh_involved' || fieldName === 'veh_one_travel_dir_code' || fieldName === 'veh_two_travel_dir_code' ) {
                        if (!crashFilterObject.hasOwnProperty('veh_two_travel_dir_code')) {
                            targetWidget.setFilter(crashFilterObject['tot_veh_involved'], crashFilterObject['veh_one_travel_dir_code'], '');
                        }
                        else if (crashFilterObject.hasOwnProperty('veh_two_travel_dir_code')) {
                            targetWidget.setFilter(crashFilterObject['tot_veh_involved'], crashFilterObject['veh_one_travel_dir_code'], crashFilterObject['veh_two_travel_dir_code']);
                        }
                    }
                    else if (targetWidget.hasOwnProperty('setFilter')) {
                        if (targetWidget.isQuickFilter == isQuickFilterMenu) {
                            targetWidget.setFilter(selectionString);
                        }
                    }
                }
            }
        }

        if (crashFilterObject.hasOwnProperty('sri')) {
            self.crashMap.display3DView();
            self.crashMap.map.sriHistogram.update(crashFilterObject, true, false);
        }
        else {
            self.crashMap.display2DView();
            self.crashMap.update();
        }
    }

    /**
     * Searches for a filter menu item or filter value based on the inputted `searchText`. 
     * If a full or partial match is found, the filter menu item or value will be highlighted.
     * @date 2024-11-21
     * @param {string} searchText The text to search in the filter menu items or values.
     * @function searchFilters
     */
    this.searchFilters = function(searchText) {
        self.clearSearchHighlights();

        for (let index = 0; index < self.filterWidgets.length; index++) {
            const widget = self.filterWidgets[index];
            if (widget.hasOwnProperty('searchFilter')) {
                const targetWidgets = widget.searchFilter(searchText);
                if (targetWidgets !== undefined) {
                    if (targetWidgets.length > 0) { 
                        targetWidgets.forEach(element => {
                            element.classList.add(self.highlightClassName);
                            if (!element.classList.contains("filter-container")) element.classList.add('animated-bg');
                        });
                    }                         
                }
            }
        }
    }

    /**
     * Clears the search highlight when searching for a filter cateogry or value.
     * @date 2024-11-21
     * @function clearSearchHighlights
     */
    this.clearSearchHighlights = function () {
        const classNamesToRemove = [self.highlightClassName, 'animated-bg'];  // Classes to find

        // Use document.querySelectorAll to select div elements with all the class names
        const elements = document.querySelectorAll(`div.${self.highlightClassName}`);

        // Loop through the selected elements
        elements.forEach(element => {
            classNamesToRemove.forEach(classNameToRemove => {
                element.classList.remove(classNameToRemove);
            });
        });
    }

    /**
     * Gets all the selected Jurisdiction filters. Returns on object of code list and description list.
     * @function getSelectedJurisdictions
     * @return {object} An object containing all selected jurisdiction filters.
     * @example
     * console.log(crashFilter.getSelectedJurisdictions);   // { codes: ['01', '02', ...], descriptions: ['Atlantic', Bergen, ...] }
     */
    this.getSelectedJurisdictions = function() {
        var selectedJuris = {
            codes: [],
            descriptions: [],
            jurisdiction: ''
        }

        self.filterWidgets.forEach((widget) => {
            if (widget.hasOwnProperty('content')) {
                if (widget.content.id) {
                    if (widget.content.id.includes('jursidictionFilterMenu')) {
                        const juriCheckboxFilter = widget.checkBoxes;
                        juriCheckboxFilter.forEach(checkBoxRow => {
                            if (checkBoxRow.checked) {
                                selectedJuris.codes.push(checkBoxRow.rowAttributes.code);
                                selectedJuris.descriptions.push(checkBoxRow.rowAttributes.description);
                                if (checkBoxRow.rowAttributes.code.length === 2) selectedJuris.jurisdiction = 'county';
                                else if (checkBoxRow.rowAttributes.code.length === 4) selectedJuris.jurisdiction = 'municipality';
                            }
                        });
                    }
                }
            }
        });

        return selectedJuris;
    }

    /**
     * Sets the input text of the Omni Search bar based on the descriptions of the selected filters.
     * @param  {object} selectedFiltersObject Format: { codes: [...], descriptions: [...] }
     * @return None. Sets the Omni Search input text.
     */
    this.setOmniSearchWithSelections = function(selectedFiltersObject) {
        const omniSearchInput = document.getElementsByClassName('omni-input');
        omniSearchInput[0].value = selectedFiltersObject.descriptions.join('; ');
    }

    /**
     * Disables jurisdiction items depending on the juri type that was initally selected.
     * If a county was selected, all muni items will be disabled and vice versa.
     * @param  {object} selectedFiltersObject Format: { codes: [...], descriptions: [...] }
     * @return None. Adds a disable class to the checkbox row.
     */
    this.disableJurisdictionItems = function(selectedFiltersObject) {
        // if selected jurisdictions are counties, disable municipality options
        self.filterWidgets.forEach((widget) => {
            if (widget.hasOwnProperty('content')) {
                if (widget.content.id) {
                    if (widget.content.id.includes('jursidictionFilterMenu')) {
                        const juriCheckboxFilter = widget.checkBoxes;
                        juriCheckboxFilter.forEach(checkBoxRow => {
                            if (!checkBoxRow.checked) {
                                if (selectedFiltersObject.jurisdiction === 'county') {
                                    if (checkBoxRow.hasOwnProperty('rowAttributes')) {
                                        if (!checkBoxRow.rowAttributes.hasOwnProperty('fieldName')) checkBoxRow.disable(true);
                                    }
                                } else if (selectedFiltersObject.jurisdiction === 'municipality') {
                                    if (checkBoxRow.hasOwnProperty('rowAttributes')) {
                                        if (checkBoxRow.rowAttributes.hasOwnProperty('fieldName')) checkBoxRow.disable(true);
                                    }
                                } else checkBoxRow.disable(false);
                            }
                        });
                    }
                }
            }
        });
    }

    /**
     * Removes the disabled class from jurisdiction checkboxrow items.
     * @return None.
     */
    this.enableJurisdictionItems = function() {
        // if selected jurisdictions are counties, disable municipality options
        self.filterWidgets.forEach((widget) => {
            if (widget.hasOwnProperty('content')) {
                if (widget.content.id) {
                    if (widget.content.id.includes('jursidictionFilterMenu')) {
                        const juriCheckboxFilter = widget.checkBoxes;
                        juriCheckboxFilter.forEach(checkBoxRow => {
                            checkBoxRow.disable(false);
                        });
                    }
                }
            }
        });
    }

    /**
     * Fires a click handler on the checkbox row of the specified jurisdiction code.
     * Mainly used by the Omni Search.
     * @param  {string} jurisdictionCode County: 2 digit; Muni: 4 digit
     * @return None. The checkbox row of the jurisdiction will be "clicked" and will fire subsequent handlers relating to the checkboxrow object.
     */
    this.selectJurisdiction = function(jurisdictionCode, fireClickHandler = false) {
        let ctyCode = jurisdictionCode.length === 2 ? jurisdictionCode : jurisdictionCode.substring(0, 2);

        self.filterWidgets.forEach((widget) => {
            if (widget.hasOwnProperty('content')) {
                if (widget.content.id) {
                    if (widget.content.id === 'jursidictionFilterMenu' + ctyCode) {
                        const juriCheckboxFilter = widget.checkBoxes;
                        for (let index = 0; index < juriCheckboxFilter.length; index++) {
                            const checkBoxRow = juriCheckboxFilter[index];
                            if (checkBoxRow.rowAttributes.code === jurisdictionCode) {
                                if (fireClickHandler) checkBoxRow.checkBox.click();
                                else checkBoxRow.uncheck();
                            }
                        }
                    }
                }
            }
        });
    }

    /**
     * Clears all selections from each filter widget.
     * @date 2024-10-30
     * @function reset
     */
    this.reset = function() {
        // unhighlight filter toggle buttons
        self.filterWidgetToggles.forEach(toggleButton => {
            toggleButton.classList.remove('green-button-toggle');
        });

        self.filterWidgets.forEach((widget) => {
            if (widget.hasOwnProperty('reset')) {
                widget.reset();
            }
        });
    }

    /**
     * Gets the filter widget based on widget ID.
     * @date 2024-10-30
     * @param {string} id The widget ID to find.
     * @returns {function} The widget module based on widget ID.
     */
    this.getWidgetById = function(id) {
        return (self.filterWidgets.find(widget => widget.id === id));
    }
}
