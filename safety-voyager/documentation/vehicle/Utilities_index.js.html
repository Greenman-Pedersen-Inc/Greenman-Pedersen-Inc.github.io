import './utilities.css';

const siteRootURL = '../login';

/**
 * Contains helper functions for widgets to call.
 * @date 2024-11-20
 * @module Utilities
 */


/**
 * Creates a Modal DOM Node which functions as a popup. The modal contains a button which can redirect the user if clicked on.
 * @date 2024-10-14
 * @module Modal
 * @param {string} text Text that is displayed in the modal body div.
 * @param {string} title The title that appears in the modal header.
 * @param {string} buttonText The text that appears in the modal button.
 * @param {boolean} redirect=false If true, redirects the user.
 * @returns {domNode} A DOM Node of the Modal div.
 * @example
 * const modal = new Modal("Saved Successfully", "Your user query was saved successfully.");
 * someContainerDiv.append(modal);
 */
function createModal(text, title = 'Error Information', buttonText = 'Accept', redirect = false) {
    const modalId = 'error-modal';
    const modalElement = document.getElementById(modalId);
    if (modalElement) modalElement.remove(); // remove any duplicate error messages

    let modal = document.createElement('div');
    modal.id = modalId;
    modal.className = 'modal fade bd-example-modal-sm show';
    modal.role = 'dialog';
    modal.style = 'display: block';

    let modalDialog = document.createElement('div');
    modalDialog.className = 'modal-dialog modal-sm modal-dialog-centered';
    modal.appendChild(modalDialog);

    let modalContent = document.createElement('div');
    modalContent.className = 'modal-content';
    modalDialog.appendChild(modalContent);

    let modalHeader = document.createElement('div');
    modalHeader.className = 'modal-header';
    modalContent.appendChild(modalHeader);

    let modalTitle = document.createElement('h5');
    modalTitle.className = 'modal-title';
    modalTitle.innerHTML = text;
    modalTitle.title = title;
    modalHeader.append(modalTitle);

    let modalBody = document.createElement('div');
    modalBody.className = 'modal-body';
    modalContent.appendChild(modalBody);

    let modalBodyText = document.createElement('p');
    modalBodyText.innerHTML = title;
    modalBody.appendChild(modalBodyText);

    let modalFooter = document.createElement('div');
    modalFooter.className = 'modal-footer';
    modalContent.appendChild(modalFooter);

    let acceptButton = document.createElement('button');
    acceptButton.className = 'btn btn-primary';
    acceptButton.type = 'button';
    acceptButton.innerHTML = buttonText;
    acceptButton.addEventListener('click', function (event) {
        if (redirect) {
            window.location = siteRootURL;
        } else {
            modal.remove();
        }
    });
    modalFooter.appendChild(acceptButton);
    document.body.style.display = 'block';
    document.body.prepend(modal);
}

/**
 * Handles errors by evaluating their type and details, then displays appropriate modal dialogs.
 *
 * @function errorHandler
 * @date 2024-11-20
 * @param {Object} error - The error object containing information about the error.
 *
 * - If the error's `httpCode` is 498 or the error name is 'identity-manager:aborted',
 *   a modal prompts the user to sign in again due to an expired session.
 * - If the error contains nested `details` with specific `httpStatus`, it handles accordingly.
 * - Otherwise, the error object is passed directly to `createModal`.
 */
function errorHandler(error, message) {
    if (error.httpCode === 498) {
        createModal(
            'Your session has expired. In order to continue using Safety Voyager you must sign in again. Closing this dialog will redirect to the login where you will be able to sign in again.',
            'Credentials Timed Out',
            true
        );
    } else if (error.name === 'identity-manager:aborted') {
        createModal(
            'Your session has expired. In order to continue using Safety Voyager you must sign in again. Closing this dialog will redirect to the login where you will be able to sign in again.',
            'Credentials Timed Out',
            true
        );
    } else if (error.details) {
        if (error.details.error) {
            if (error.details.error.details) {
                if (error.details.error.details.httpStatus === 498) {
                    createModal(
                        'Your session has expired. In order to continue using Safety Voyager you must sign in again. Closing this dialog will redirect to the login where you will be able to sign in again.',
                        'Credentials Timed Out',
                        'Accept',
                        true
                    );
                } else {
                    createModal(error);
                }
            }
        } else {
            createModal(error);
        }
    } else {
        createModal(error);
    }
}

/**
 * Evaluates the visibility of an element based on the current zoom level and specified minimum and maximum zoom constraints.
 *
 * @function evaluateZoomVisibility
 * @date 2024-11-20
 * @param {number} zoomLevel - The current zoom level to evaluate.
 * @param {number} minzoom - The minimum zoom level for visibility (optional).
 * @param {number} maxzoom - The maximum zoom level for visibility (optional).
 * @returns {boolean} `true` if the zoom level is within the specified range or if no constraints are provided, otherwise `false`.
 */
function evaluateZoomVisibility(zoomLevel, minzoom, maxzoom) {
    if (minzoom && maxzoom) {
        if (zoomLevel >= minzoom && zoomLevel <= maxzoom) {
            return true;
        } else {
            return false;
        }
    } else if (minzoom) {
        if (zoomLevel >= minzoom) {
            return true;
        } else {
            return false;
        }
    } else if (maxzoom) {
        if (zoomLevel <= maxzoom) {
            return true;
        } else {
            return false;
        }
    } else {
        return true;
    }
}

String.prototype.toTitleCase = function () {
    return this.replace(/\w\S*/g, function (txt) {
        return txt.charAt(0).toUpperCase() + txt.substr(1).toLowerCase();
    });
};

export default { evaluateZoomVisibility, errorHandler, createModal };
