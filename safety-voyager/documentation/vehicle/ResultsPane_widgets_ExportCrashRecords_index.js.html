import { Modal } from 'bootstrap';
import API from '../../../API';
import AgreementModal from './widgets/AgreementModal';

/**
 * Creates the Export Crash Records button div and appends it to the div `toggleAttachPoint`.
 * Handles exporting crash record data based on the crashes on the map or from an array of crash record IDs `crashIDArray`. 
 * @date 2024-11-21
 * @module ResultsPane/widgets/ExportCrashRecords
 * @param {domNode} toggleAttachPoint The div where the Export Crash Records button will be appended to.
 * @param {module} crashMap The Crash Map module.
 * @param {domNode} mapContainer The div containing the Mapbox map.
 * @param {string[]|undefined} crashIDArray=undefined An array of crash IDs to get crash record information for.
 * If `undefined`, the exported records will be based on the crashes in the map view.
 * @requires bootstrap
 * @requires ../../../API
 * @requires ./widgets/AgreementModal
 */
export default function ExportCrashRecords(toggleAttachPoint, crashMap, mapContainer, crashIDArray = undefined) {
    const self = this;
    const api = new API();
    const recordLimit = 50000;
    const agreementModal = new AgreementModal(mapContainer, self);
    const toggleButton = document.createElement('div');
    toggleButton.title = 'Export crash record data in current view; limited to 50,000 records';
    toggleButton.className = 'omni-element ri-folder-zip-line';
    toggleButton.addEventListener('click', function () {
        self.enableSpinner(true);
        if (crashIDArray) {
            exportRecords(crashIDArray);
        } else {
            exportRecords();
        }
    });
    toggleAttachPoint.append(toggleButton);

    toggleButton.addEventListener('mouseenter', function () {
        if (!toggleButton.classList.contains('green-button-toggle')) {
            toggleButton.classList.add('orange-button-toggle');
        }
    });
    toggleButton.addEventListener('mouseleave', function () {
        toggleButton.classList.remove('orange-button-toggle');
    });

    /**
     * Toggles a spinner on a button to indicate an active or inactive state. 
     * Replaces the button's icon with a spinner when enabled and restores the original icon when disabled.
     *
     * @function enableSpinner
     * @date 2024-11-20
     * @param {boolean} enable=true - Determines whether to enable the spinner. Defaults to `true`.
     */
    this.enableSpinner = function (enable = true) {
        if (enable) {
            toggleButton.classList.remove('ri-folder-zip-line');
            toggleButton.classList.add('export-spinner');
        } else {
            toggleButton.classList.remove('export-spinner');
            toggleButton.classList.add('ri-folder-zip-line');
        }
    };

    /**
     * Displays the agreement modal after clicking on the toggle button.
     * @date 2022-02-05
     * @param {string} url - Blob object response from the crashRecordExport route.
     * @param {int} exportCount - Number of rows in the exported file.
     * @returns None.
     */
    function displayAgreementModal(url, exportCount) {
        agreementModal.update(url, exportCount);
        var agreementModalObj = new Modal(agreementModal.content, {
            keyboard: false,
            backdrop: false
        });
        agreementModalObj.show();
    }

    /**
     * Exports crash records based on a provided array of crash IDs or the currently displayed map view. 
     * If no `crashIDArray` is provided, it collects crash records from the cluster view on the map. 
     * Handles various scenarios such as record limits, empty data, and server responses.
     *
     * @function exportRecords
     * @date 2024-11-20
     * @param {string[]|undefined} crashIDArray=undefined - An optional array of crash IDs to export. 
     * If not provided, crash records will be derived from the map's cluster view.
     *
     * @returns {void} This function handles exporting records and displaying relevant alerts or modals.
     *
     * @example
     * exportRecords(['12345', '67890']); // Export specific crash records
     * exportRecords(); // Export crash records from the current map view
     */
    function exportRecords(crashIDArray = undefined) {
        if (crashIDArray || (crashMap.isCrashView() && crashMap.map.sriHistogram.disabled)) {
            if (crashIDArray === undefined) {
                // this is on the cluster view. Concat all the crash_arrays from each cluster feature
                const features = crashMap.map.queryRenderedFeatures({ layers: ['cluster_circles'] });
                if (features.length > recordLimit) {
                    new Modal({
                        title: 'No Crash Records Found',
                        content: 'Please alter your map view to display 50,000 crash records or.',
                        buttonText: 'Continue'
                    }).show();
                    return;
                }
                else if (features.length > 0) {
                    // Initialize an empty array to store the combined result
                    crashIDArray = [];

                    features.forEach(feature => {
                        if (feature.properties.hasOwnProperty('crash_array')) {
                            const parsedIdArray = JSON.parse(feature.properties.crash_array);
                            crashIDArray.push(...parsedIdArray);
                        }
                    });
                } else {
                    alert('No crash records were detected in the current map view.');
                    return;
                }
            }
            api.getExportRecordsByID(JSON.stringify(crashIDArray)).then((response) => {
                if (response.status === 200) {
                    if (response.statusCode) {
                        alert(
                            'There was an issue exporting crash records. Please refresh your map view and try again.'
                        );
                    } else if (response.headers.get('exportCount') <= 0) {
                        alert(
                            `No crashes exported from the current map view. Please change the map view and try again.`
                        );
                        return;
                    } else {
                        response
                            .blob()
                            //response.json()
                            .then((blob) => URL.createObjectURL(blob))
                            .then((url) => {
                                displayAgreementModal(url, response.headers.get('exportCount'));
                            });
                    }
                } else {
                    alert('There was an issue exporting crash records. Please refresh your map view and try again.');
                }
            }).finally(() => {
                self.enableSpinner(false);
            });
        } 
        else {
            // of user is exporting from the county or muni level, query records from the cty/muni features from the results grid
            let exportQuery = null;
            let featureSet = new Set();
            const crashFilterObject = crashMap.crashFilter.getSelectedFilters();

            if (exportQuery !== null) {
                var fetchParams = {
                    headers: {
                        credentials: JSON.stringify(crashMap.credentials)
                    },
                    cache: 'no-cache' // *default, no-cache, reload, force-cache, only-if-cached
                };
                fetch(exportQuery, fetchParams).then((response) => {
                    if (response.status === 200) {
                        if (response.statusCode) {
                            alert(
                                'There was an issue exporting crash records. Please refresh your map view and try again.'
                            );
                        } else if (response.headers.get('exportCount') <= 0) {
                            alert(
                                `No crashes exported from the current map view. Please change the map view and try again.`
                            );
                            return;
                        } else {
                            response
                                .blob()
                                .then((blob) => URL.createObjectURL(blob))
                                .then((url) => {
                                    displayAgreementModal(url, response.headers.get('exportCount'));
                                });
                        }
                    } else {
                        alert(
                            'There was an issue exporting crash records. Please refresh your map view and try again.'
                        );
                    }
                }).finally(() => {
                    self.enableSpinner(false);
                });
            }
            else {
                alert(
                    'There was an issue exporting crash records. Please refresh your map view and try again.'
                );            
            }

            return;
        }
    }
}
