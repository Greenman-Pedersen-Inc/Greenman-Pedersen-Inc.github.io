export default function TopLocations(crashMap, crashFilter, defaultTableOption) {
    const self = this;

                    demogidx_5: row.properties.demogidx_5,
                    lowincpct: row.properties.lowincpct,
                };
            });

            self.grid.gridOptions.api.setRowData(gridData);

    /**
     * Highlights a feature based on the target hover item in the results table.
     * @date 2022-02-14
     * @param {object} event - mouseover event object from ag-grid
     * @returns None. Adds a geometry shape to the map using CrashMap/widgets/HighlightFeature
     */
    function highlightFeature(event) {
        if (event.data) {
            crashMap.map.crashSegmentsHeatmap.highlightFeature(event.data);
        }
    }

    function removeHighlight(event) {
        if (event.data) {
            if (event.data.hasOwnProperty('segment_id')) {
                crashMap.map.highlightFeature.clearHighlight();
            }
        }
    }

    return this;

/**
 * Returns a completed promise of the top locations data based on filters
 * @date 3/15/2024 - 9:05:10 AM
 *
 * @async
 * @param {object} selectedFilters selected filters from the filters menu
 * @returns {object}
 */
async function getTopLocationsData(selectedFilters, grid) {
    const filters = {
        option_select: 'county',
        selected_filters: JSON.stringify(selectedFilters),
    };

    let requestPromise = api.getTopLocations(filters);
    requestPromise.then((data) => {
        console.log(data);
        const colList = processFeatures(data);
        grid.gridOptions.api.setColumnDefs(colList);
        grid.gridOptions.api.setRowData(data);
        grid.gridOptions.api.sizeColumnsToFit();
    });
}

function processFeatures(dataObj) {
    function processColumnList(dataObj) {
        let columns = [];
        // process columns list
        if (dataObj.length > 0) {
            const firstRow = dataObj[0];
            Object.keys(firstRow).forEach((key) => {
                let colItem = {
                    field: key,
                    resizeable: true,
                    unSortIcon: true,
                    suppressSizeToFit: true,
                    minWidth: 150,
                };
                if (isNumeric(firstRow[key])) {
                    colItem.valueFormatter = function (params) {
                        if (params.value) {
                            return `${params.value.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ',')}`;
                        }
                    };
                    colItem.cellDataType = 'number';
                }
                columns.push(colItem);
            });
        }
        return columns;
    }

    function isNumeric(value) {
        return !isNaN(parseFloat(value)) && isFinite(value);
    }

    const colList = processColumnList(dataObj);
    console.log(colList);
    return colList;
}
