import API from '../../../../API';
import SaveQueryInterface from '../SaveQueryInterface';
import MoveablePopup from './../../../../MoveablePopup';
import mapboxgl from 'mapbox-gl';
import $ from 'jquery';

import './interface.css';
import 'bootstrap/dist/css/bootstrap.min.css';

/**
 * Creates the User Query interface. 
 * Displays the user's saved queries and handles the map and filter updates when a user applies a saved query.
 * @date 2024-11-22
 * @param {domNode} attachPoint The div where the User Queries interface will be attached to.
 * @param {module} toggleButton The User Queries module.
 * @param {module} crashMap The Crash Map module.
 * @param {module} crashFilter The Crash Filter module.
 * @param {module} resultsPane The Results Pane module.
 * @requires ../../../../API
 * @requires ../SaveQueryInterface
 * @requires ../../../../MoveablePopup
 * @requires mapbox-gl
 * @requires jquery
 * @requires ./interface.css
 * @requires bootstrap/dist/css/bootstrap.min.css
 */
export default function Interface(attachPoint, toggleButton, crashMap, crashFilter, resultsPane) {
    const self = this;
    const api = new API();
    this.userName = crashMap.credentials.username;
    self.queryItems = []; // holds all the of the user query objects
    self.selectedQueryIndex; // the index of the current selected user query item

    this.domNode = document.createElement('div');
    this.domNode.id = 'userQueriesInterface';
    this.domNode.className = 'user-queries-dialog';

    const modalContentDiv = document.createElement('div');
    modalContentDiv.classList = 'modal-content user-queries-interface';
    this.domNode.append(modalContentDiv);

    // Header Div
    const header = document.createElement('div');
    header.className = 'header';
    header.innerHTML = 'User Queries';

    const headerText = document.createElement('span');

    const closeButton = document.createElement('div');
    closeButton.className = 'close ri-close-line';
    closeButton.title = 'Close';
    closeButton.addEventListener('click', function (event) {
        self.close();
    });

    header.append(headerText, closeButton);
    modalContentDiv.append(header);

    // Modal Body Content
    const modalBodyDiv = document.createElement('div');
    modalBodyDiv.classList = 'modal-body modal-body-text modal-body-padding';
    modalContentDiv.append(modalBodyDiv);

    // User Query Interface Contents
    const contentRow = document.createElement('div');
    contentRow.classList = 'row';
    modalBodyDiv.append(contentRow);

    const savedQueriesColumn = document.createElement('div');
    savedQueriesColumn.classList = 'col-4';
    contentRow.append(savedQueriesColumn);
    const savedQueriesTitle = document.createElement('h6');
    savedQueriesTitle.innerHTML = 'Saved Queries';
    savedQueriesColumn.append(savedQueriesTitle);

    // List Group - query items will be listed in this div
    const savedQueriesList = document.createElement('div');
    savedQueriesList.classList = 'list-group list-group-div';
    savedQueriesList.id = 'savedQueriesList';
    savedQueriesList.setAttribute('role', 'tablist');
    savedQueriesColumn.append(savedQueriesList);

    const queryDescriptionsCol = document.createElement('div');
    queryDescriptionsCol.classList = 'col-8';
    contentRow.append(queryDescriptionsCol);
    const queryDescriptionsTitle = document.createElement('h6');
    queryDescriptionsTitle.innerHTML = 'Query Filters';
    const queryDescriptionsDiv = document.createElement('div');
    queryDescriptionsDiv.classList = 'query-description-div';
    queryDescriptionsCol.append(queryDescriptionsTitle, queryDescriptionsDiv);

    // list group content - query descriptions will be listed in this div
    const queryDescriptionsContent = document.createElement('div');
    queryDescriptionsContent.classList = 'tab-content query-description-tab-control';
    queryDescriptionsContent.id = 'savedQueriesDescriptions';
    queryDescriptionsDiv.append(queryDescriptionsContent);

    // Modal Footer Div
    const modalFooterDiv = document.createElement('div');
    modalFooterDiv.classList = 'modal-footer query-modal-footer';

    // Modal Footer Delete Button
    const modalFooterDeleteButton = document.createElement('button');
    modalFooterDeleteButton.classList = 'action-button btn btn-sm button-margin-right';
    modalFooterDeleteButton.type = 'button';
    modalFooterDeleteButton.innerHTML = 'Delete';
    modalFooterDeleteButton.addEventListener('click', function () {
        if (self.selectedQueryIndex !== undefined) {
            const queryObject = self.queryItems[self.selectedQueryIndex];
            const title = queryObject['filters_title'];
            if (confirm(`Do you want to delete ${queryObject['filters_title']}?`)) {
                const deleteQuery = api.deleteUserQueries(self.userName, queryObject['oid']);
                const fetchParams = {
                    method: 'DELETE',
                    cache: 'no-cache', // *default, no-cache, reload, force-cache, only-if-cached
                    headers: {
                        credentials: JSON.stringify(crashMap.credentials)
                    }
                };
                fetch(deleteQuery, fetchParams).then((response) => {
                    if (response.status === 200 && response.statusCode === undefined) {
                        response.json().then((resp) => {
                            if (resp.success) {
                                self.updateMessageText(`${title} successfully deleted.`, true);
                                self.refresh(self.userName);
                            } else {
                                self.updateMessageText(`${title} could not be deleted.`, false);
                            }
                        });
                    }
                });
            }
        }
    });

    // Modal Footer Save Button
    const modalFooterSaveButton = document.createElement('button');
    modalFooterSaveButton.classList = 'action-button btn btn-sm';
    modalFooterSaveButton.type = 'button';
    modalFooterSaveButton.innerHTML = 'Save New Query';
    modalFooterSaveButton.addEventListener('click', function () {
        self.displayNewQueryInterface();
    });

    // Modal Footer Run Button
    const modalFooterRunButton = document.createElement('button');
    modalFooterRunButton.classList = 'action-button btn btn-sm float-end';
    modalFooterRunButton.type = 'button';
    modalFooterRunButton.innerHTML = 'Run';
    modalFooterRunButton.addEventListener('click', function () {
        if (self.selectedQueryIndex !== undefined) {
            crashFilter.reset();
            const queryObject = self.queryItems[self.selectedQueryIndex];
            const crashFiltersObject = JSON.parse(queryObject['filters']);
            applyMapSettings(queryObject);
            resultsPane.updateFilterSummary(queryObject['filters_description']); // set the filter descriptions in the Results Pane
            crashFilter.setFilters(crashFiltersObject, queryObject['quick_filter_menu']);
        }
    });

    // Confirmation message text
    const messageDiv = document.createElement('div');
    messageDiv.classList = 'float-end padding-top-6';
    messageDiv.id = `messageDiv`;

    // add the Save User Query Interface
    const saveQueryInterface = new SaveQueryInterface(self, modalBodyDiv, crashMap, crashFilter);

    modalContentDiv.append(modalFooterDiv);
    modalFooterDiv.append(modalFooterDeleteButton, modalFooterSaveButton, modalFooterRunButton, messageDiv);
    saveQueryInterface.attachButtons(modalFooterDiv); // attach Save Query interface buttons

    // ===== FUNCTIONS =====
    /**
     * Closes the interface by removing the DOM node, toggling the button off, 
     * and clearing its associated interface reference.
     * @function close
     */
    this.close = function () {
        self.domNode.remove();
        toggleButton.toggleOff();
        toggleButton.interface = undefined;
    };

    /**
     * Displays the interface by un-hiding the DOM node, toggling the button on,
     * and populating user queries.
     * @function show
     */
    this.show = function () {
        emptyQueriesList();
        self.domNode.classList.remove('hidden');
        toggleButton.toggleOn();
        getUserQueries(self.userName);
    };

    /**
     * Checks if the interface is currently open.
     * @function isOpen
     * @returns {boolean} Returns `true` if the interface is visible, `false` otherwise.
     */
    this.isOpen = function () {
        return !self.domNode.classList.contains('hidden');
    };

    /**
     * Empties the list of saved queries and their descriptions by removing all child nodes.
     */
    function emptyQueriesList() {
        function removeAllChildNodes(parent) {
            while (parent.firstChild) {
                parent.removeChild(parent.firstChild);
            }
        }
        removeAllChildNodes(savedQueriesList);
        removeAllChildNodes(queryDescriptionsContent);
    }

    this.refresh = function (userName) {
        emptyQueriesList();
        getUserQueries(self.userName);
    };

    /**
     * Fetches and populates the user's saved queries.
     */ 
    function getUserQueries(userName) {
        const getterQuery = api.getUserQueries(self.userName);
        var fetchParams = {
            headers: {
                cache: 'no-cache', // *default, no-cache, reload, force-cache, only-if-cached
                credentials: JSON.stringify(crashMap.credentials)
            }
        };
        fetch(getterQuery, fetchParams).then((response) => {
            if (response.status === 200 && response.statusCode === undefined) {
                response.json().then((queryItems) => {
                    self.queryItems = queryItems;
                    addListItems(queryItems);
                });
            }
        });
    }

    /**
     * creates a list group item and content for a list of query items
     * @date 2022-02-06
     * @param {object} queryItem - `
     *                  filters: "{sri:00000515__,occupant_phys_cond_killed:1,occupant_phys_cond_incapacitated:1,year:2015,2016,2017,2018,2019}"
                        filters_description: "SRI: 00000515__; Year: (2015, 2016, 2017, 2018, 2019); Occupants: (Fatality, Incapacitated)"
                        filters_title: "SFM Sri range"
                        max_mp: "12.5"
                        min_mp: "0"
                        oid: "47"
                        results_text: "ROUTE 515 (00000515__)"
                        rotation: null
                        user_name: "snopachinda@gpinet.com"
                        x_max: null
                        x_max_merc: "-7798886.463402763"
                        x_min: null
                        x_min_merc: "-8822531.146197792"
                        y_max: null
                        y_max_merc: "5205268.577923799"
                        y_min: null
                        y_min_merc: "4629239.1327667385"
                        zoom_level: "0"`
     * @returns None.
     */
    function addListItems(queryItems) {
        if (queryItems.length === 0) {
            const listGroupContent = document.createElement('div');
            listGroupContent.classList = 'tab-pane fade show active';
            listGroupContent.innerHTML = "Add a query by clicking the 'Save New Query' button below!";
            queryDescriptionsContent.append(listGroupContent);
        }

        for (let index = 0; index < queryItems.length; index++) {
            const queryItem = queryItems[index];

            const listGroupItem = document.createElement('a');
            listGroupItem.classList = `${
                index === 0 ? 'list-group-item list-group-item-action active' : 'list-group-item list-group-item-action'
            }`;
            listGroupItem.id = `list-${index}-list`;
            listGroupItem.href = '#list-' + index;
            listGroupItem.innerHTML = queryItem['filters_title'];
            listGroupItem.setAttribute('data-bs-toggle', 'list');
            listGroupItem.setAttribute('role', 'tab');
            listGroupItem.setAttribute('aria-controls', 'list-' + index);
            listGroupItem.setAttribute('index', index);
            listGroupItem.addEventListener('click', function () {
                self.selectedQueryIndex = index;
            });
            if (index === 0) self.selectedQueryIndex = index;
            savedQueriesList.append(listGroupItem);

            const listGroupContent = document.createElement('div');
            listGroupContent.classList = `${index === 0 ? 'tab-pane fade show active' : 'tab-pane fade'}`;
            listGroupContent.id = 'list-' + index;
            var filterDescriptionObject = JSON.parse(queryItem['filters_description']);
            for (const [key, value] of Object.entries(filterDescriptionObject)) {
                listGroupContent.innerHTML += `<b> ${key} </b>: ${value} <br>`;
            }
            listGroupContent.setAttribute('role', 'tabpanel');
            listGroupContent.setAttribute('aria-labelledby', `list-${index}-list`);
            queryDescriptionsContent.append(listGroupContent);
        }
    }

    /**
     * Hides the action buttons in the modal footer.
     *
     * @function hideRunButtons
     * @description Adds the 'hidden' class to the run, delete, and save buttons in the modal footer.
     * @date 2024-12-16
     */
    function hideRunButtons() {
        modalFooterRunButton.classList.add('hidden');
        modalFooterDeleteButton.classList.add('hidden');
        modalFooterSaveButton.classList.add('hidden');
    }

    /**
     * Shows the action buttons in the modal footer.
     *
     * @function showRunButtons
     * @description Removes the 'hidden' class from the run, delete, and save buttons in the modal footer.
     * @date 2024-12-16
     */
    function showRunButtons() {
        modalFooterRunButton.classList.remove('hidden');
        modalFooterDeleteButton.classList.remove('hidden');
        modalFooterSaveButton.classList.remove('hidden');
    }

/**
 * Updates the message text displayed in the message div.
 *
 * @function updateMessageText
 * @description Displays a confirmation or error message with a fade-in and fade-out effect.
 * @param {string} text - The message text to display.
 * @param {boolean} isSuccess - Indicates whether the message is a success (true) or failure (false).
 * @date 2024-12-16
 */
    this.updateMessageText = function (text, isSuccess) {
        // the confirmation message will dissapear after 5 seconds
        setTimeout(function () {
            $('#messageDiv').fadeOut('fast').empty();
        }, 5000); // <-- time in milliseconds
        messageDiv.classList.remove('hidden');
        messageDiv.innerHTML = text;
        if (isSuccess) {
            messageDiv.classList.remove('failText');
            messageDiv.classList.add('successText');
        } else {
            messageDiv.classList.remove('successText');
            messageDiv.classList.add('failText');
        }
        $('#messageDiv').fadeIn();
    };

/**
 * Displays the new query interface and hides the content row.
 *
 * @function displayNewQueryInterface
 * @description Hides the main content row, shows the query interface, and hides action buttons.
 * @date 2024-12-16
 */
    this.displayNewQueryInterface = function () {
        contentRow.classList.add('hidden');
        hideRunButtons();
        saveQueryInterface.show();
    };

/**
 * Hides the new query interface and refreshes the main view.
 *
 * @function hideNewQueryInterface
 * @description Hides the query interface, refreshes the content, and shows the main content row with action buttons.
 * @date 2024-12-16
 */
    this.hideNewQueryInterface = function () {
        saveQueryInterface.hide();
        self.refresh(self.userName);
        showRunButtons();
        contentRow.classList.remove('hidden');
    };

/**
 * Applies map settings based on a query object.
 *
 * @function applyMapSettings
 * @description Adjusts the map's bounds, zoom level, pitch, and rotation according to the query parameters.
 * @param {Object} queryObject - Object containing map settings (x_min, x_max, y_min, y_max, zoom_level, pitch, rotation).
 * @date 2024-12-16
 */
    function applyMapSettings(queryObject) {
        const minX = parseFloat(queryObject['x_min']);
        const maxX = parseFloat(queryObject['x_max']);
        const minY = parseFloat(queryObject['y_min']);
        const maxY = parseFloat(queryObject['y_max']);
        const mapBounds = new mapboxgl.LngLatBounds(new mapboxgl.LngLat(minX, maxY), new mapboxgl.LngLat(maxX, minY));
        const options = {
            maxZoom: parseFloat(queryObject['zoom_level']),
            pitch: parseFloat(queryObject['pitch']),
            bearing: parseFloat(queryObject['rotation']),
            linear: true,
            duration: 0
        };
        // const newCameraTransform = crashMap.map.cameraForBounds(mapBounds, options);
        // crashMap.map.jumpTo(newCameraTransform);
        crashMap.map.fitBounds(mapBounds, options);
    }

    new MoveablePopup(attachPoint, self.domNode, header, 'center');
    self.refresh(self.userName);
}
