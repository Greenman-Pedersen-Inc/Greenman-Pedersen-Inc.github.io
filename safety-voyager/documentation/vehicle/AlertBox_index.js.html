AlertBox/index.js

import '../voyager.widgets.css';
import './alert.box.css';

/**
 * Determines the age of an update and returns a boolean if the update exceeds the default number of days since the update.
 * @function calculateUpdateAge
 * @param {number} daysSinceUpdate The maximum number of days to display the update message. 
 * @returns {boolean} Returns true if the data has been updated in the time frame.
 */
function calculateUpdateAge(daysSinceUpdate = 7) {
    const today = new Date();
    const updateDate = updates.dateSaved;
    const nextWeek = updateDate.setDate(updateDate.getDate() + daysSinceUpdate);

    if (nextWeek < Date.parse(today)) {
        return true;
    } else {
        return false;
    }
}

/**
 * This widget informs the user of the latest changes to the data.
 * If the data has been updated in the last 7 days it will add an alert to the icon.
 * A script file called updatesContent.js stores information about the update.
 * The file data structure and functionality could be updated to include multiple updates.
 * @module AlertBox
 * @requires ../voyager.widgets.css
 * @requires ./alert.box.css
 * @param {domNode} toggleAttachPoint The parent div to attach the Alert Box to.
 * @param {domNode} filterContents The div containing the filter contents.
 * @param {object[]} dataUpdates An array of objects containing update messages.
 * @param {function} crashMap The CrashMap module.
 * @example
 * const mapContainer = document.getElementById('map-container'); 
 * const filterContents = document.getElementById('filter-contents');
 * const dataUpdates = [{title: "New Data Update", text: "Data was updated on..."}, ...];
 * const alertBox = new AlertBox(mapContainer, filterContents, dataUpdates, crashMap);
 */
export default function AlertBox(toggleAttachPoint, filterContents, dataUpdates, crashMap) {
    var timer;
    const self = this;
    crashMap.AlertBox = this;
    const updatesBoxHeader = document.createElement('div');
    const updatesBoxContent = document.createElement('div');
    const filterControls = document.createElement('div');
    filterControls.className = 'mapboxgl-ctrl mapboxgl-ctrl-group';

    this.content = document.createElement('div');
    this.content.className = `omni-element updates-box has-title hidden`;

    updatesBoxHeader.className = 'omni-title';
    updatesBoxHeader.innerHTML = 'Safety Voyager Updates';

    updatesBoxContent.className = 'updates-box-content';

    if (dataUpdates && dataUpdates.length > 0) {
        dataUpdates.forEach((update) => {
            let updateContent = document.createElement('div');
            let updateHeader = document.createElement('div');

            updateHeader.className = 'update-header';
            updateHeader.innerHTML = update.title ? update.title : '';

            updateContent.className = 'update-content';
            updateContent.innerHTML = update.text ? update.text : '';

            updatesBoxContent.append(updateHeader, updateContent);
        });
    }

    this.toggle = document.createElement('div');
    this.toggle.className = 'omni-element ri-alarm-warning-line';
    this.toggle.title = 'Display Updates';
    this.toggle.id = 'alertsButton';
    this.toggle.addEventListener('click', function (event) {
        // hide any filter menus or settings menu
        const filterToggles = document.getElementById('filter-toggles');
        for (var i = 0; i < filterToggles.children.length; i++) {
            filterToggles.children[i].classList.add('hidden');
        }

        const filterContainers = document.getElementById('filter-contents');
        for (var i = 0; i < filterContainers.children.length; i++) {
            if (!filterContainers.children[i].classList.contains('updates-box')) {
                filterContainers.children[i].classList.add('hidden');
            }
        }
        document.getElementById('settingsButton').classList.remove('green-button-toggle');
        self.content.classList.toggle('hidden');

        if (self.toggle.classList.contains('orange-button-toggle'))
            self.toggle.classList.remove('orange-button-toggle');
        self.toggle.classList.toggle('green-button-toggle');

        // if there are OmniSearch results, hide them as they block the menu
        document.querySelector('.search-omni-box-results').classList.add('hidden');
    });
    this.toggle.addEventListener('mouseenter', function () {
        if (!self.toggle.classList.contains('green-button-toggle')) {
            self.toggle.classList.add('orange-button-toggle');
        }
    });
    this.toggle.addEventListener('mouseleave', function () {
        self.toggle.classList.remove('orange-button-toggle');
    });
    this.hide = function () {
        self.content.classList.add('hidden');
        self.toggle.classList.remove('green-button-toggle');
    };
    this.show = function () {
        self.content.classList.remove('hidden');
        self.toggle.classList.add('green-button-toggle');
    };
    this.content.append(updatesBoxHeader, updatesBoxContent);

    filterControls.append(this.toggle);
    toggleAttachPoint.append(filterControls);
    filterContents.append(this.content);
}
