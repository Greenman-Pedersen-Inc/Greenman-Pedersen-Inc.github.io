import DatePicker from '../../../../../../../node_modules/vanillajs-datepicker/js/Datepicker';
import '../../../../../../../node_modules/vanillajs-datepicker/dist/css/datepicker-bs5.min.css';
import './date.range.selector.css';

/**
 * Creates buttons for the date range filters and defines toggle functionalites for date range buttons.
 * @date 2024-11-20
 * @module FilterMenu/widgets/DateSelector/widgets/DateRangeSelector
 * @param {domNode} attachPoint The div where the month selector will be appended to.
 * @param {domNode} parent The parent div of the month selector.
 * @param {string} attribute The code name for this filter type.
 * @param {number} endYear The end year.
 * @requires ../../../../../../../node_modules/vanillajs-datepicker/js/Datepicker
 * @requires ../../../../../../../node_modules/vanillajs-datepicker/dist/css/datepicker-bs5.min.css
 * @requires ./date.range.selector.css
 */
export default function DateRangeSelector(attachPoint, parent, attribute, endYear) {
    const self = this;
    this.attribute = attribute;
    const startDate = new Date(endYear, 0, 1);
    const endDate = new Date(endYear, 11, 31);
    // var customRow = domConstruct.create('tr', null, attachPoint);
    const customRow = document.createElement('tr');
    attachPoint.append(customRow);

    const rowLabel = document.createElement('td');
    rowLabel.className = 'row-label custom-row';
    rowLabel.innerHTML = 'R';
    rowLabel.tile = 'Clear Date Range filter';
    rowLabel.setAttribute('colspan', 2);
    rowLabel.addEventListener('click', function (event) {
        self.clearDateRangeSelection(true);
    });

    const startTimeBox = document.createElement('td');
    startTimeBox.className = 'custom-row';
    startTimeBox.setAttribute('colspan', 3);

    const endTimeBox = document.createElement('td');
    endTimeBox.className = 'custom-row';
    endTimeBox.setAttribute('colspan', 3);
    customRow.append(rowLabel, startTimeBox, endTimeBox);

    const dateStartHolder = document.createElement('input');
    startTimeBox.append(dateStartHolder);

    const dateEndHolder = document.createElement('input');
    endTimeBox.append(dateEndHolder);

    /**
     * Clears the selected date range and resets the related UI elements.
     *
     * @function clearDateRangeSelection
     * @date 2024-11-20
     */
    this.clearDateRangeSelection = function (update) {
        dateStartHolder.value = formatDateInput(startDate);
        dateEndHolder.value = formatDateInput(endDate);

        self[attribute] = undefined; // delete all year values in filter object
        dateStartHolder.classList.remove('green-button-toggle');
        startTimeBox.classList.remove('green-button-toggle');
        dateEndHolder.classList.remove('green-button-toggle');
        endTimeBox.classList.remove('green-button-toggle');
    };

    /**
     * Formats a given Date object into a string in the format MM/DD/YYYY.
     *
     * @param {Date} d - The date to format.
     * @returns {string} A string representing the formatted date in MM/DD/YYYY format.
     *
     * @example
     * const date = new Date(2024, 10, 20); // November 20, 2024
     * console.log(formatDateInput(date)); // "11/20/2024"
     */
    function formatDateInput(d) {
        var curr_date = d.getDate();
        var curr_month = d.getMonth() + 1; //Months are zero based
        var curr_year = d.getFullYear();
        return `${curr_month}/${curr_date}/${curr_year}`;
    }

    /**
     * Gets the selected filter attributes and codes from the date range selector widget.
     * @function getSelections
     * @date 2024-11-20
     * @returns {Object|undefined} An object containing the selected date filters.
     */
    this.getSelections = function () {
        if (self.hasOwnProperty(attribute)) {
            if (self[attribute] !== undefined) {
                console.log(self[attribute]);
                if (self[attribute].length > 0) {
                    return {
                        [attribute]: self[attribute]
                    };
                }
            }
        }
    };

    /**
     * Gets the human-readble descriptions from the date range selector widget.
     * @date 2024-11-19
     * @returns {object} An object containing the selected date filters formatted in human-readble descriptions.
     */
    this.getDescriptionSelections = function () {
        if (self.hasOwnProperty(attribute)) {
            if (self[attribute] !== undefined) {
                if (self[attribute].length > 0) {
                    return { 'Date Range': self[attribute] };
                }
            }
        }
    };

    /**
     * Gets the date range selector widget if the type `filterField` matches this widget.
     * @date 2024-11-19
     * @function checkFilterField
     * @param {string} filterField The type of date widget to access.
     * @returns {module} The Date Range Selector widget.
     */
    this.checkFilterField = function (filterField) {
        if (self.attribute === filterField) {
            return this;
        }
    };

    /**
     * @deprecated
     */
    this.searchFilters = function(searchText) {

    }

    /**
     * Sets the date range based on the filterValueString
     * @date 2022-03-15
     * @param {string} filterValueString - the value of date_range filter: "2020-01-01;2020-04-29"
     * @returns None. Sets this widget's attribute.
     */
    this.setFilter = function (filterValueString) {
        const parsedDateString = filterValueString.split(';');
        dateStart.setDate(new Date(parsedDateString[0]));
        dateEnd.setDate(new Date(parsedDateString[1]));
    };

    /**
     * Returns a formatted date range string in the format "YYYY-MM-DD;YYYY-MM-DD".
     *
     * @function returnRange
     * @date 2024-11-20
     * @param {string|Date} dateStart - The start date, either as a string or a Date object.
     * @param {string|Date} dateEnd - The end date, either as a string or a Date object.
     * @returns {string} A string representing the formatted date range in the format "YYYY-MM-DD;YYYY-MM-DD",
     *                   or an empty string if the start date is after the end date.
     *
     * @example
     * const range = obj.returnRange('2024-11-01', '2024-11-20');
     * console.log(range); // "2024-11-01;2024-11-20"
     *
     * @example
     * const invalidRange = obj.returnRange('2024-12-01', '2024-11-20');
     * console.log(invalidRange); // ""
     */
    this.returnRange = function (dateStart, dateEnd) {
        const start = new Date(dateStart);
        const end = new Date(dateEnd);

        function FormatDateString(d) {
            var curr_date = d.getDate().toString();
            var curr_month = (d.getMonth() + 1).toString(); //Months are zero based
            var curr_year = d.getFullYear();
            return curr_year + '-' + curr_month.padStart(2, '0') + '-' + curr_date.padStart(2, '0');
        }
        if (start <= end) {
            return FormatDateString(start) + ';' + FormatDateString(end);
        } else {
            return '';
        }
    };

    const dateStart = new DatePicker(dateStartHolder, {
        buttonClass: 'btn',
        maxDate: formatDateInput(endDate),
        autohide: true,
        orientation: 'left',
        defaultViewDate: startDate
    });
    var dateEnd = new DatePicker(dateEndHolder, {
        buttonClass: 'btn',
        maxDate: formatDateInput(endDate),
        autohide: true,
        orientation: 'left',
        defaultViewDate: endDate
    });
    dateStartHolder.value = formatDateInput(startDate);
    dateEndHolder.value = formatDateInput(endDate);

    dateStartHolder.addEventListener('changeDate', function (evt) {
        if (evt.detail.date.getTime() === startDate.getTime()) {
            dateStartHolder.classList.remove('green-button-toggle');
            startTimeBox.classList.remove('green-button-toggle');
        } else {
            dateStartHolder.classList.add('green-button-toggle');
            startTimeBox.classList.add('green-button-toggle');
            self[attribute] = self.returnRange(dateStart.picker.viewDate, dateEnd.picker.viewDate);
            // parent.toggle.classList.add('green-button-toggle');
        }
    });

    dateEndHolder.addEventListener('changeDate', function (evt) {
        if (evt.detail.date.getTime() === endDate.getTime()) {
            dateEndHolder.classList.remove('green-button-toggle');
            endTimeBox.classList.remove('green-button-toggle');
        } else {
            dateEndHolder.classList.add('green-button-toggle');
            endTimeBox.classList.add('green-button-toggle');
            self[attribute] = self.returnRange(dateStart.picker.viewDate, dateEnd.picker.viewDate);
            // parent.toggle.classList.add('green-button-toggle');
        }
    });
}
