import API from '../../../API';
import MapSources from '../MapSources';
import mapboxgl from 'mapbox-gl';
function orderMagnitudeStops(routeData) {
    const redMin = 0;
    const greenMin = 0;
    const blueMin = 0;
    const redMax = 255;
    const greenMax = 255;
    const blueMax = 255;
    const steps = [];

    // USING HEATMAP COLORS
    if (routeData.segments > 1) {
        // when dividing the difference it should be at least 1
        if (routeData.max_crashes - routeData.min_crashes > 2) {
            steps.push([routeData.min_crashes, `rgb(${redMin}, ${greenMax}, ${blueMin})`]);
            steps.push([
                (routeData.max_crashes - routeData.min_crashes) / 2,
                `rgb(${redMax}, ${greenMax}, ${blueMin})`
            ]);
            steps.push([routeData.max_crashes, `rgb(${redMax}, ${greenMin}, ${greenMin})`]);
        }
        // don't create a redundant entry in the legend
        else if (routeData.max_crashes === routeData.min_crashes) {
            steps.push([routeData.min_crashes, `rgb(${redMin}, ${greenMax}, ${blueMin})`]);
        } else {
            steps.push([routeData.min_crashes, `rgb(${redMin}, ${greenMax}, ${blueMin})`]);
            steps.push([routeData.max_crashes, `rgb(${redMax}, ${greenMin}, ${greenMin})`]);
        }
    } else if (routeData.segments > 0) {
        steps.push([routeData.max_crashes, `rgb(${redMax}, ${greenMin}, ${blueMin})`]);
    }

    return steps;
}

function orderMagnitudeStopsHeight(map, routeData, layer) {
    const steps = [];
    const baseHeight = map.getPaintProperty(layer, 'fill-extrusion-base') + 10;
    steps.push([routeData.min_crashes, baseHeight]);
    if (routeData.segments > 300) {
        steps.push([routeData.max_crashes, 10000]);
    } else if (routeData.segments > 100) {
        steps.push([routeData.max_crashes, 2500]);
    } else if (routeData.segments > 50) {
        steps.push([routeData.max_crashes, 1250]);
    } else if (routeData.segments > 0) {
        steps.push([routeData.max_crashes, 500]);
    }
    return steps;
}

function fillExtrusionHeight(parent, map, routeData, layer, attribute) {
    this.createColorRamp = function () {
        let colorRamp = [];
        newFillExtrusionColor.stops.forEach((stop) => {
            colorRamp.push({
                from: stop[0],
                to: stop[0],
                color: stop[1]
            });
        });
        return colorRamp;
    };
    let newFillExtrusionColor = {};
    let newfillExtrusionHeight = {};
    if (routeData.segments && routeData.segments > 0) {
        newFillExtrusionColor.property = attribute;
        newFillExtrusionColor.stops = orderMagnitudeStops(routeData);
        newfillExtrusionHeight.property = attribute;
        newfillExtrusionHeight.stops = orderMagnitudeStopsHeight(map, routeData, layer);
    } else {
        newFillExtrusionColor.property = attribute;
        newFillExtrusionColor.stops = [[0, 'rgb(100, 100, 100)']];
        newfillExtrusionHeight.property = attribute;
        newfillExtrusionHeight.stops = [[0, 0]];
    }
    map.setPaintProperty(layer, 'fill-extrusion-color', newFillExtrusionColor);
    map.setPaintProperty(layer, 'fill-extrusion-height', newfillExtrusionHeight);
    parent.extrusionHeight = newfillExtrusionHeight;
}

function updateVisualization(parent, crashMap, routeInformation, mapRotate) {
    // sometimes the milepost information comes back weird
    // try catch should help narrow it down a bit
    try {
        const startMPPoint = JSON.parse(routeInformation.start_mp_point);
        const endMPPoint = JSON.parse(routeInformation.end_mp_point);
        const routeBounds = new mapboxgl.LngLatBounds(startMPPoint.coordinates, endMPPoint.coordinates);
        const theta =
            (Math.atan2(
                endMPPoint.coordinates[0] - startMPPoint.coordinates[0],
                endMPPoint.coordinates[1] - startMPPoint.coordinates[1]
            ) *
                180) /
            Math.PI;

        // if no crash features are selected, this attribute will return null as a result of the query structure from the API
        crashMap.map.crashFeatures = routeInformation.crash_features || [];

        // console.log(routeInformation.start_mp_point, routeInformation.end_mp_point);
        if (routeInformation.start_mp_point === routeInformation.end_mp_point) {
            crashMap.map.easeTo({ center: startMPPoint.coordinates, zoom: 15 }, { preserveVisiblity: true });
        } else {
            // move the map viewer to the extents of the route section selected
            if (mapRotate) {
                const newCameraTransform = crashMap.map.cameraForBounds(routeBounds, {
                    padding: 200,
                    bearing: theta - 90
                });
                newCameraTransform.pitch = 60;

                crashMap.map.easeTo(newCameraTransform, { preserveVisiblity: true });
            }
        }

        parent.rotationView = {
            bounds: routeBounds,
            pitch: 60,
            bearing: theta - 90,
            padding: 200
        };
    } catch (error) {
        console.log(error);
    } finally {
        // set route symbology
        const layerSymbology = new fillExtrusionHeight(
            parent,
            crashMap.map,
            routeInformation,
            'route_histogram_layer',
            'crash_count'
        );

        crashMap.resultsPane.updateFilterSummary(); // ResultsPane is attached in the ResultsPane widget on initalization
        parent.legendCard.update(layerSymbology.createColorRamp());
    }
}

export default function SRIHistogram(crashMap, map, crashFilter, layerGroupID) {
    const self = this;
    const api = new API();
    const mapSources = new MapSources(map);
    const layerGroup = mapSources.getLayerGroupData(layerGroupID);
    const appliedFilters = crashFilter.getSelectedFilters();
    this.rotationView = null;
    this.extrusionHeight; // used for the histogram highlight layer so the highlight bar has the correct height
    this.loadingIndicators = [];
    Object.defineProperty(this, 'disabled', {
        get: function () {
            return this['_disabled'];
        },
        set: function (value) {
            this['_disabled'] = value;
            if (value) {
                layerGroup.layers.forEach(function (layer) {
                    map.setFilter(layer.id, null);
                    map.getLayer(layer.id).disabled = value;
                    map.setLayoutProperty(layer.id, 'visibility', 'none');
                });
            } else {
                layerGroup.layers.forEach(function (layer) {
                    map.getLayer(layer.id).disabled = value;
                    map.setLayoutProperty(layer.id, 'visibility', 'visible');
                });
            }
        },
        enumerable: false
    });

    // Adds the SRI histogram source to the map. Should be called only once as the source layer gets updated instead of being deleted and re-created.
    this.addToMap = function (appliedFilters, disabled = false) {
        layerGroup.layers.forEach(function (layer) {
            if (!map.getSource(layer.source) && layer['source-tiles']) {
                map.addSource(layer.source, {
                    type: 'vector',
                    tiles: [layer['source-tiles'](appliedFilters).tileEndpoint],
                    attribution: 'New Jersey Department of Treasury NJTR-1 Reports 2020'
                });
            }
            if (layer.beforeLayer) {
                map.addLayer(layer, layer.beforeLayer);
            } else {
                map.addLayer(layer);
            }
            // map.on('sourcedata', initializeLayer);
            if (layer.mouseleave) {
                map.on('mouseleave', layer.id, layer.mouseleave);
            } else {
                map.on('mouseenter', layer.id, function () {
                    map.getCanvas().style.cursor = 'pointer';
                });
            }
            if (layer.mouseenter) {
                map.on('mouseenter', layer.id, layer.mouseenter);
            } else {
                map.on('mouseleave', layer.id, function () {
                    map.getCanvas().style.cursor = '';
                });
            }
        });

        self.disabled = disabled;
        if (!self.legendCard) {
            // only the symbology for the histogram will change so the other components of the card are not included.
            const layerDefinition = mapSources.getMapLayerData('route_histogram_layer');
            layerDefinition.relatedLayers = ['route_unmatched_layer', 'sri_crash_label'];  // array to store related layers that share the Legend toggle
            self.legendCard = map.legend.addLayer(layerDefinition, [], 'visible', true, true, true);
        }
    };

    /**
     * Updates the SRI histogram with the selected filters and SRI. Fills out the SRI name and MP inputs in the Location filter widget.
     * @param  {object[]} selectedFiltersObject From crashFilter.getSelectedFilters(). Format: { sri: "...", crash_type: "..." }
     * @param  {string[]} sriName Human-readble SRI name with SRI code. Format: 'NJ 24 (0000000024__)'
     * @return None. Generates 3D SRI histogram and zooms to it. Opens the Location filter widget.
     */
    this.update = function (appliedFilters, updateUnmatchedRoute = true, mapRotate = false) {
        // start updating the route information, this promise will be picked up in the update source function
        self.getRouteInformation = api.getRouteCrashData(appliedFilters, crashMap.credentials.token);
        self.mapRotate = mapRotate;
        layerGroup.layers.forEach(function (layer) {
            if (map.getSource(layer.source) && layer['source-tiles']) {
                if (layer.id != 'route_unmatched_layer') {
                    const histogramQuery = layer['source-tiles'](appliedFilters);
                    map.getSource(layer.source).setTiles([histogramQuery.tileEndpoint + `&dt=${Date.now()}`]);
                } else if (updateUnmatchedRoute) {
                    // if updateUnmatchedRoute is false, don't update the unmatched route extrusion (grey line)
                    const unmatchedQuery = layer['source-tiles'](appliedFilters);
                    map.getSource(layer.source).setTiles([unmatchedQuery.tileEndpoint + `&dt=${Date.now()}`]);
                }
            }
        });
    };

    this.updateSource = function (source) {
        if (source.sourceId === layerGroup.source) {
            self.legendCard.loadingIndicator.classList.remove('hidden');
            crashMap.FilterMenu.isLoading(true);

            if (map.isSourceLoaded(source.sourceId)) {
                const appliedFilters = crashFilter.getSelectedFilters();
                const locationFilterWidget = crashFilter.getWidgetById('LocationFilter');

                if (appliedFilters.sri) {
                    self.getRouteInformation
                        .then((routeInformation) => {
                            if (routeInformation) {
                                // if mp_start has not been established, set the Location filter widget MP data
                                //     locationFilterWidget.extentSelector.setMilepostBoxes(
                                //         routeInformation.start_mp,
                                //         routeInformation.end_mp
                                //     );
                                
                                locationFilterWidget.extentSelector.setAbsoluteMinMaxExtents(
                                    routeInformation.absolute_start_mp,
                                    routeInformation.absolute_end_mp
                                );

                                if (locationFilterWidget.extentSelector.checkMilepostBoxes()) {
                                    locationFilterWidget.extentSelector.setMilepostBoxes(
                                        routeInformation.absolute_start_mp,
                                        routeInformation.absolute_end_mp
                                    );                                    
                                }

                                // if (!appliedFilters.hasOwnProperty('mp_end')) {

                                //     locationFilterWidget.extentSelector.setAbsoluteMinMaxExtents(
                                //         routeInformation.absolute_start_mp,
                                //         routeInformation.absolute_end_mp
                                //     );
                                // }

                                updateVisualization(self, crashMap, routeInformation, self.mapRotate);

                                self.routeInformation = routeInformation;
                                self.mapRotate = false;
                            }
                        })
                        .catch((error) => {
                            console.log(error);
                        })
                        .finally(() => {
                            self.legendCard.loadingIndicator.classList.add('hidden');
                            self.loadingIndicators.forEach((indicator) => {
                                indicator.classList.remove('loading');
                            });
                            crashMap.FilterMenu.isLoading(false);
                            crashMap.FilterMenu.toggle.highlight();
                            crashMap.FilterMenu.clearCheckBoxSpinners();
                        });
                }
            }
        }
    };

    /**
     * Rotates the map to the SRI bounds and pitch
     * @date 2022-04-21
     */
    this.resetView = function () {
        if (self.rotationView) {
            crashMap.map.easeTo({
                pitch: self.rotationView.pitch,
                bearing: self.rotationView.bearing,
                padding: self.rotationView.padding
            });
        }
    };

    this.updateHighlight = function (milepost) {
        const features = crashMap.map.queryRenderedFeatures({
            layers: ['route_histogram_layer'],
            filter: ['in', 'mp', milepost]
        });
        const labelSourceName = `${layerGroupID}_highlight_source`;
        const labelLayerName = `${layerGroupID}_highlight`;
        const source = map.getSource(labelSourceName);
        var histogramFeatures = {
            features: [],
            type: 'FeatureCollection'
        };
        if (features && features.length > 0) {
            features.forEach((feature) => {
                histogramFeatures.features.push(feature);
            });
        }
        if (source) {
            crashMap.map.setPaintProperty(labelLayerName, 'fill-extrusion-height', self.extrusionHeight);
            source.setData(histogramFeatures);
        } else {
            map.addSource(labelSourceName, {
                type: 'geojson',
                data: histogramFeatures
            });
            crashMap.map.addLayer({
                id: labelLayerName,
                type: 'fill-extrusion',
                source: labelSourceName,
                layout: {
                    visibility: 'visible'
                },
                paint: {
                    'fill-extrusion-base': 20.1,
                    'fill-extrusion-color': 'black',
                    'fill-extrusion-opacity': 0.5,
                    'fill-extrusion-vertical-gradient': true,
                    'fill-extrusion-height': self.extrusionHeight
                }
            });
        }
    };

    this.removeHighlight = function () {
        const labelSourceName = `${layerGroupID}_highlight_source`;
        const source = map.getSource(labelSourceName);
        if (source) {
            source.setData({
                features: [],
                type: 'FeatureCollection'
            });
        }
    };

    this.isDisabled = function () {
        return self.disabled;
    }

    this.addToMap(appliedFilters, true);
    map.on('sourcedata', self.updateSource);
}
