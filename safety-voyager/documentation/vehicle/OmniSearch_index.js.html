OmniSearch/index.js

import SearchFilterButton from './widgets/SearchFilterButton';
import OmniResult from './widgets/OmniResult';
import urls from '../urls';

import '../voyager.widgets.css';
import './omni.search.css';

/**
 * Creates the search box element and handles the results UI when a user types a search prompt.
 * @date 2024-11-20
 * @module OmniSearch
 * @param {module} crashFilter The Crash Filter module.
 * @param {module} crashMap The Crash Map module.
 * @param {module} resultsPane The Results Pane module.
 * @param {module} filterMenu The Filter Menu module.
 * @param {module} settingsMenu The Settings Menu module.
 * @param {domNode} container The div where the Omni Search widget will be appended to.
 * @param {object[]} searchTypes An array containing objects of search options. 
 * @requires ./widgets/SearchFilterButton
 * @requires ./widgets/OmniResult
 * @requires ../urls
 * @requires ../voyager.widgets.css
 * @requires ./omni.search.css
 * @example
 * const searchTypes = [
    {
        className: 'ri-hashtag signals-section-control-number-filter',
        toolTip: 'Search for a signal section control number.',
        disabled: false
    }, {...}, ,,,];
 */
export default function OmniSearch(
    crashFilter,
    crashMap,
    resultsPane,
    filterMenu,
    alertBox,
    settingsMenu,
    container,
    searchTypes
) {
    const self = this;
    const maxSearchResults = 20;
    crashMap.omniSearch = self;
    this.searchFilterButtons = [];
    /**
     * The main container for the OmniSearch bar
     */
    const searchOmniBox = document.createElement('div');
    searchOmniBox.className = 'omni-container';
    searchOmniBox.title = 'Enter Route Number For a list of Selections';

    /**
     * The text input bar
     */
    const searchOmniInput = document.createElement('input');
    searchOmniInput.className = 'omni-input';
    searchOmniInput.placeholder = 'Search by Route # / Address / Place';
    searchOmniInput.autocomplete = 'off';
    searchOmniInput.addEventListener('mouseenter', function (event) {
        if (searchOmniInput.value !== '') {
            searchOmniInputClear.classList.remove('hidden');
        }
    });
    searchOmniInput.addEventListener('mouseleave', function (event) {
        if (event.toElement && !event.toElement.classList.contains('omni-input-clear')) {
            searchOmniInputClear.classList.add('hidden');
        }
    });

    /**
     * Event listener for text input. Generates a list of search results based on text input and selected search options.
     */
    searchOmniInput.addEventListener('keyup', function (event) {
        const self = this;
        filterMenu.hideAll(); // hide the filter menu as it is obstructs the search results
        if (event.code === 'ArrowDown') {
            if (document.querySelector('.searchOmniBoxResults').classList.contains('hidden')) {
            } else {
                this.blur();
                // I want to be able to arrow through the search results, hit enter and go to the result
                //document.getElementsByClassName('searchOmniBoxResultElement')[0].focus()
            }
        } else {
            if (this.value && this.value !== '') {
                const filter = {
                    searchText: this.value,
                    includeRoute: document.querySelector('.state-route-filter')
                        ? document.querySelector('.state-route-filter').classList.contains('green-button-toggle')
                        : false,
                    includeCounty: document.querySelector('.county-filter')
                        ? document.querySelector('.county-filter').classList.contains('green-button-toggle')
                        : false,
                    includeMunicipality: document.querySelector('.municipality-filter')
                        ? document.querySelector('.municipality-filter').classList.contains('green-button-toggle')
                        : false,
                    includeCaseNumber: document.querySelector('.case-number-filter')
                        ? document.querySelector('.case-number-filter').classList.contains('green-button-toggle')
                        : false,
                    includeAddress: document.querySelector('.google-place-filter')
                        ? document.querySelector('.google-place-filter').classList.contains('green-button-toggle')
                        : false
                };
                searchOmniInput.classList.add('loading');

                document.querySelector('.search-hanger').classList.remove('hidden');

                clearTimeout(self.timeoutTracker);
                const fetchUrl =
                    urls.searchURL +
                    Object.keys(filter)
                        .map((key) => key + '=' + filter[key])
                        .join('&');
                var fetchParams = {
                    headers: {
                        credentials: JSON.stringify(crashMap.credentials)
                    }
                };

                self.timeoutTracker = setTimeout(function () {
                    fetch(fetchUrl, fetchParams).then((response) => {
                        searchOmniInput.classList.remove('loading');
                        searchOmniBoxSearchType.classList.add('square-bottom-right');
                        if (response.status === 200) {
                            searchOmniBoxResults.removeResults();
                            response.json().then((searchDataList) => {
                                const searchData = searchDataList['SearchResults'];
                                if (searchData.length > 0) {
                                    searchOmniBoxResults.show();
                                }
                                for (let index = 0; index < searchData.length; index++) {
                                    if (index >= maxSearchResults) {
                                        break;
                                    }
                                    const searchResult = searchData[index];
                                    new OmniResult(
                                        crashFilter,
                                        crashMap,
                                        resultsPane,
                                        searchResult,
                                        searchOmniInput,
                                        searchOmniBoxResults,
                                        searchOmniBox
                                    );
                                }
                            });
                        } else {
                            console.log('There was an issue with requesting data from route Search:' + response);
                        }
                    });
                }, 1000);
            } else {
                searchOmniBoxResults.removeResults();
                searchOmniBoxResults.hide();
            }
        }
    });

    /**
     * Event listener when the input bar is clicked. Shows the results list if there are items in it. Hides the filter menu
     */
    searchOmniInput.addEventListener('click', function (event) {
        if (document.querySelector('.search-omni-box-results').children.length > 0) {
            searchOmniBoxResults.show();
        }
        filterMenu.hideAll();
        alertBox.hide();
        settingsMenu.hide();
    });

    const searchOmniInputClear = document.createElement('div');
    searchOmniInputClear.className = 'omni-input-clear hidden';

    searchOmniInputClear.addEventListener('click', function (event) {
        searchOmniInputClear.classList.add('hidden');
        filterMenu.crashFilter.resetWidget.reset();
        searchOmniBoxResults.removeResults();
    });

    /**
     * Button div to open the search options menu.
     */
    const searchOmniBoxSearchType = document.createElement('div');
    searchOmniBoxSearchType.className = 'omni-element omni-search ri-equalizer-line green-button-toggle';
    searchOmniBoxSearchType.id = 'omni-search-options';
    searchOmniBoxSearchType.title = 'Toggle Search Categories';
    searchOmniBoxSearchType.addEventListener('click', function () {
        searchOmniBoxSearchType.classList.remove('orange-button-toggle');

        self.highlightToggleButton();

        if (searchTypeOptions.classList.contains('hidden')) {
            searchHanger.classList.remove('hidden');
            searchTypeOptions.classList.remove('hidden');
            searchOmniBoxSearchType.classList.add('square-bottom-right');
        } else {
            searchTypeOptions.classList.add('hidden');
        }
    });

    searchOmniBoxSearchType.addEventListener('mouseenter', function () {
        if (!searchOmniBoxSearchType.classList.contains('green-button-toggle')) {
            searchOmniBoxSearchType.classList.add('orange-button-toggle');
        }
    });
    searchOmniBoxSearchType.addEventListener('mouseleave', function () {
        searchOmniBoxSearchType.classList.remove('orange-button-toggle');
    });

    /**
     * Div that is underneath the input bar. Houses the search results list.
     */
    const searchHanger = document.createElement('div');
    searchHanger.className = 'search-hanger';

    /**
     * Div that houses the search options.
     */
    const searchTypeOptions = document.createElement('div');
    searchTypeOptions.className = 'search-type-options hidden';
    searchTypeOptions.hide = function () {
        this.classList.add('hidden');
    };
    searchTypeOptions.show = function () {
        this.classList.remove('hidden');
    };
    /**
     * Div that houses the returned search results.
     */
    const searchOmniBoxResults = document.createElement('div');
    searchOmniBoxResults.className = 'search-omni-box-results hidden';
    searchOmniBoxResults.removeResults = function () {
        while (this.firstChild) {
            this.removeChild(this.firstChild);
        }
    };
    searchOmniBoxResults.hide = function () {
        this.classList.add('hidden');

        searchOmniBoxSearchType.classList.remove('square-bottom-right');
    };
    searchOmniBoxResults.show = function () {
        this.classList.remove('hidden');
        searchOmniBoxSearchType.classList.add('square-bottom-right');
    };
    /**
     * Generate the search options in the options div
     */
    searchTypes.forEach(function (filterAttributes) {
        const searchFilterButon = new SearchFilterButton(searchTypeOptions, filterAttributes, self);
        self.searchFilterButtons.push(searchFilterButon);
    });

    /**
     * Append elements to their parent divs
     */
    searchOmniBox.append(searchOmniInput, searchOmniBoxSearchType, searchOmniInputClear);
    searchHanger.append(searchOmniBoxResults);
    container.append(searchOmniBox, searchHanger, searchTypeOptions);

    /**
     * Toggles the visibility of the results list and options menu.
     * @date 2024-11-20
     * @function hide
     * @param {boolean} hide Hides the results list and options menu
     */
    this.hide = function (hide) {
        if (hide) {
            searchOmniBoxResults.hide();
            searchTypeOptions.hide();
        } else {
            searchOmniBoxResults.show();
            searchTypeOptions.show();
        }
    };

    /**
     * Checks if any of the search filter buttons are toggled by verifying 
     * the presence of the 'green-button-toggle' class.
     * @function isSearchFilterButtonToggled
     * @date 2024-11-20
     * @returns {boolean} `true` if at least one button is toggled, otherwise `false`.
     */
    this.isSearchFilterButtonToggled = function () {
        for (let index = 0; index < self.searchFilterButtons.length; index++) {
            const btn = self.searchFilterButtons[index];
            if (btn.classList.contains('green-button-toggle')) return true;
        }
        return false;
    };

    /**
     * Generates search results based on a custom query or specific crash filters.
     * @date 2024-11-20
     * @param {object} attributes The currently selected crash filters.
     * @function fireCustomQuery
     */
    this.highlightToggleButton = function () {
        if (self.isSearchFilterButtonToggled()) searchOmniBoxSearchType.classList.add('green-button-toggle');
        else searchOmniBoxSearchType.classList.remove('green-button-toggle');
    };

    /**
     * Generates search results based on a custom query or specific crash filters.
     * @date 2024-11-20
     * @param {object} attributes The currently selected crash filters.
     * @function fireCustomQuery
     */
    this.fireCustomQuery = function (attributes) {
        const query = new OmniResult(
            crashFilter,
            crashMap,
            resultsPane,
            attributes,
            searchOmniInput,
            searchOmniBoxResults,
            searchOmniBox
        );
        query.domNode.click();
    };
}
