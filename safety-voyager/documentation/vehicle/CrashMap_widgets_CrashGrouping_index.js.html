/**
 * Creates the Crash Details popup which displays a table of crashes within a crash cluster or network segment.
 * The user can click on a crash in the table to display the Crash Record Information dialog.
 * Renders pie charts to display a breakdown of crashes by cluster, route, and state grouped by a crash filter.
 * @date 2024-10-31
 * @module CrashMap/widgets/CrashGrouping
 * @requires ../../../MoveablePopup
 * @requires ag-grid-community
 * @requires chart.js
 * @requires ../../../API
 * @requires ../CrashRecordDialog
 * @requires ./widgets/BreakdownComparator
 * @requires ./../../../FeedbackForm
 * @requires ../../../ResultsPane/widgets/ExportCrashRecords
 * @requires ag-grid-community/dist/styles/ag-grid.css
 * @requires ag-grid-community/dist/styles/ag-theme-alpine.css
 * @requires ./crash.grouping.css
 */

import { Grid } from 'ag-grid-community';
import { Chart, PieController, ArcElement, PointElement, LinearScale, Title, CategoryScale } from 'chart.js';
import API from '../../../API';
import CrashRecordDialog from '../CrashRecordDialog';
import BreakdownComparator from './widgets/BreakdownComparator';
import FeedbackForm from './../../../FeedbackForm';
import ExportCrashRecords from '../../../ResultsPane/widgets/ExportCrashRecords';

Chart.register(PieController, ArcElement, PointElement, LinearScale, Title, CategoryScale);

import './../../../../../node_modules/ag-grid-community/styles/ag-grid.css';
import './../../../../../node_modules/ag-grid-community/styles/ag-theme-alpine.css';
import './crash.grouping.css';

/**
 * Adds HTML option elements to a combo box `attachPoint`.
 * @date 2024-10-31
 * @param {object[]} selectValues An array of objects that contain the select options of a combo box.
 * @param {domNode} attachPoint The DOM elements to attach the `selectValues` to.
 * @example
 * const selectValues = [{ label: 'Crash Type', value: 'crash_type', displayField: 'Crash_Type' }, ...];
 * const comboBoxEl = document.getElementById('crash-combobox');
 * addSelectValues(selectValues, comboBoxEl);
 */
function addSelectValues(selectValues, attachPoint) {
    const fragment = document.createDocumentFragment();

    selectValues.forEach((option) => {
        let newOption = document.createElement('option');
        newOption.value = option.value;
        newOption.innerHTML = option.label;

        fragment.append(newOption);
    });

    attachPoint.append(fragment);
}

/**
 * Removes previously opened popups or dialogs if they are displayed in the map interface.
 * @date 2024-10-31
 */
function removePriorWindows() {
    const previousWindows = document.querySelectorAll('.crash-grouping');
    previousWindows.forEach((aWindow) => {
        aWindow.remove();
    });
    const recordDialogs = document.querySelectorAll('.crash-record-dialog');
    recordDialogs.forEach((aWindow) => {
        aWindow.remove();
    });

    document.querySelectorAll('.crash-grouping').forEach((element) => {
        element.remove();
    });
}

/**
 * Creates the Crash Details popup which displays a table of crashes within a crash cluster or network segment.
 * @date 2024-10-31
 * @function CrashGrouping
 * @param {function}  crashFilter  Handles getters and setters when the user selects the crash filters.
 * @param {string[] | undefined} crashIDArray An array of crash IDs to get crash attributes for.
 * If defined, the Crash Details popup will display crash records from the array. If `undefined`, the dialog will query crash records based on the selected filters. 
 * @param {string[] | undefined} sriArray An array of SRI values.
 * If defined, the Crash Details popup will display the SRI breakdown chart based on the SRIs in the array. If `undefined`, the dialog will not render the SRI breakdown chart. 
 * @param {function} crashMap The CrashMap module.
 * @param {object} additionalFilters={} An object containing the currently selected crash filters.
 * @param {function} feature=undefined The Mapbox feature selected from the map interface.
 */
export default function CrashGrouping(crashFilter, crashIDArray, sriArray, crashMap, additionalFilters = {}, feature = undefined) {
    const api = new API();

    class FeedbackFormRenderer {
        // gets called once before the renderer is used
        init(params) {
            // create the cell
            this.eGui = document.createElement('div');
            this.eGui.innerHTML = `
                <span>
                    <button class="crash-grouping-button btn btn-sm">Submit Feedback</button>
                </span>
             `;

            // get references to the elements we want
            this.eButton = this.eGui.querySelector('.crash-grouping-button');

            // add event listener to button
            this.eButton.addEventListener('click', function (e) {
                e.stopPropagation();
                new FeedbackForm('Crash Cluster Breakdown', crashFilter, crashMap, params.data.crashid);
            });
        }

        getGui() {
            return this.eGui;
        }

        // gets called whenever the cell refreshes
        refresh(params) {
            // return true to tell the grid we refreshed successfully
            return true;
        }

        // gets called when the cell is removed from the grid
        destroy() {
            // do cleanup, remove event listener from button
            if (this.eButton) {
                // check that the button element exists as destroy() can be called before getGui()
                this.eButton.removeEventListener('click', this.eventListener);
            }
        }
    }

    // CrashGrouping: displays SRI data and relevant case
    // numbers for a specific milepost. Also breaks down data into categories via
    // pie charts.
    removePriorWindows(); // remove other Crash Grouping windows

    const self = this;
    const selectValues = [
        { label: 'Crash Type', value: 'crash_type', displayField: 'Crash_Type' },
        { label: 'Road Median Type', value: 'road_median_code', displayField: 'Road_Median_Type' },
        { label: 'Year', value: 'year', displayField: 'Year' },
        { label: 'Week Day', value: 'acc_dow', displayField: 'Day_of_Week' },
        { label: 'Environmental Condition', value: 'environ_cond_code', displayField: 'Environmental_Condition' },
        { label: 'Road Characteristic', value: 'road_char_code', displayField: 'Road_Characteristic' },
        { label: 'Road Surface Condition', value: 'surf_cond_code', displayField: 'Road_Surface_Condition' },
        { label: 'Department Type', value: 'dept_num', displayField: 'Police_Department_Type' },
        { label: 'Ramp Direction', value: 'ramp_direction', displayField: 'Ramp_Direction' },
        { label: 'Light Condition', value: 'light_cond_code', displayField: 'Light_Condition' }
    ];

    this.domNode = document.createElement('div');
    this.domNode.className = 'crash-grouping';

    // Create the div with class "card-header" and add text "Featured"
    let cardHeaderDiv = document.createElement('div');
    cardHeaderDiv.className = 'card-header fw-semibold';
    cardHeaderDiv.textContent = 'Detailed Crash Information';
    cardHeaderDiv.id = 'crash-details-header';

    const chartDiv = document.createElement('div');
    chartDiv.className = 'chart';

    const breakdownComparator = new BreakdownComparator(chartDiv);

    const functionalityDiv = document.createElement('div');
    functionalityDiv.className = 'functionality';

    const crashVariableSelector = document.createElement('select');
    crashVariableSelector.className = 'crash-variable-selector';
    crashVariableSelector.addEventListener('change', function (event) {
        const breakdownField = this.value;
        const selectedFilters = crashFilter.getSelectedFilters();
        let dataRequests = getBreakdownData(breakdownField, selectedFilters);

        breakdownComparator.update(dataRequests, breakdownField);
    });

    const spacer = document.createElement('div');
    spacer.className = 'spacer';

    functionalityDiv.append(crashVariableSelector, spacer);

    const tableDiv = document.createElement('div');
    tableDiv.className = 'table-container ag-theme-alpine';
    tableDiv.id = 'crash-grouping-table';

    self.domNode.append(cardHeaderDiv, chartDiv, functionalityDiv, tableDiv);

    // if there is a single SRI selected in the cluster, use it to get route information.
    // otherwise they will enumerated in the route breakdown section.
    /**
     * Creates the URL requests to get crash count statistics grouped by a crash category `breakdownField`.
     * @date 2024-10-31
     * @param {string} breakdownField
     * @returns {string[]} An array of URl requests that get crash count statistics grouped by a crash category `breakdownField`.
     */
    function getBreakdownData(breakdownField) {
        let selectedFilters = crashFilter.getSelectedFilters();
        for (let key in additionalFilters) {
            if (!selectedFilters.hasOwnProperty(additionalFilters)) {
              selectedFilters[key] = additionalFilters[key];
            }
        }
        const sriValues = new Set(sriArray ? sriArray : []);
        const dataRequests = [];

        if (self.resolvedFeatureData) {
            dataRequests.push(Promise.resolve(self.resolvedFeatureData));
        } 
        else if (crashIDArray === undefined) {
            dataRequests.push(api.getCrashList(selectedFilters));
        }
        else {
            dataRequests.push(api.getCrash(JSON.stringify(crashIDArray), selectedFilters.year));
        }

        if (sriValues.size === 1 || selectedFilters.sri) {
            if (!selectedFilters.hasOwnProperty('sri')) {
                selectedFilters.sri = sriValues.values().next().value;
            }
            dataRequests.push(api.getSRIBreakdown(selectedFilters, breakdownField, crashMap.credentials.token));
        } else {
            dataRequests.push(Promise.resolve({ message: sriValues }));
        }

        dataRequests.push(api.getStateBreakdown(selectedFilters, breakdownField, crashMap.credentials.token));

        return dataRequests;
    }

    /**
     * Creates the URL requests to get crash count statistics grouped by a crash category `breakdownField`.
     * @date 2024-10-31
     * @param {string} breakdownField
     * @returns {string[]} An array of URl requests that get crash count statistics grouped by a crash category `breakdownField`.
     */
    function updateTitle(resolvedFeatureData) {
        const resolvedSummaryText = `Selected Cluster ${resolvedFeatureData ? `(${resolvedFeatureData.length} Crashes)` : ''}`;
        const headerDiv = document.getElementById('crash-details-header');

        if (feature !== undefined) {
            if (feature.properties.hasOwnProperty('sri')) headerDiv.innerHTML = `SRI: ${feature.properties.sri} (Milepost ${feature.properties.milepost_range_text}) - ${resolvedSummaryText}`;
            else headerDiv.innerHTML = resolvedSummaryText;
        } else {
            headerDiv.innerHTML = resolvedSummaryText;
        }
    }

    /**
     * Renders the ag-grid to fit the container div on startup.
     * @date 2024-10-31
     * @param {function} params the ag-grid parameters function.
     */
    function onFirstDataRendered(params) {
        params.api.sizeColumnsToFit();
    }

    this.gridOptions = {
        columnDefs: [
            {
                field: 'crashid',
                headerName: 'Crash ID',
                hide: true
            },
            {
                field: 'acc_case',
                headerName: 'Case Number',
                minWidth: 175,
                sortable: true,
                filter: true
            },
            {
                field: 'year',
                headerName: 'Year',
                sortable: true,
                filter: true
            },
            {
                field: 'acc_dow',
                headerName: 'Day',
                sortable: true,
                filter: true
            },
            {
                field: 'sri',
                headerName: 'SRI',
                sortable: true,
                filter: true
            },
            {
                field: 'milepost',
                headerName: 'Milepost',
                sortable: true,
                filter: true
            },
            // {
            //     field: 'feedbackForm',
            //     minWidth: 175,
            //     cellRenderer: FeedbackFormRenderer
            // }
        ],
        rowData: [],
        onCellClicked: function (event) {
            new CrashRecordDialog(event.data.crashid, event.data.year, crashMap.credentials);
            crashMap.StatsPane.update(additionalFilters);
        },
        onFirstDataRendered: onFirstDataRendered,
        suppressCellFocus: true,
        onGridReady: (params) => {
            params.api.showLoadingOverlay();
        },
        autoSizeStrategy: {
            type: "fitGridWidth"
        },
    };

    this.grid = new Grid(tableDiv, self.gridOptions);



    let dataRequests = getBreakdownData(selectValues[0].value);

    Promise.all(dataRequests).then((data) => {
        const resolvedFeatureData = data[0];

        updateTitle(resolvedFeatureData);
        // to use for cluster breakdown when a different crash attribute is selected
        self.resolvedFeatureData = resolvedFeatureData;

        breakdownComparator.update(data, crashVariableSelector.value);

        self.gridOptions.api.setRowData(resolvedFeatureData);
        self.gridOptions.columnApi.autoSizeAllColumns();
        self.gridOptions.api.sizeColumnsToFit();

        if (crashIDArray === undefined) {
            crashIDArray = resolvedFeatureData.map(crash => crash.crashid);
        }

        let exportCrashRecords = new ExportCrashRecords(
            functionalityDiv,
            crashMap,
            document.getElementById('map-container'),
            crashIDArray
        );
    });

    addSelectValues(selectValues, crashVariableSelector);

    const crashDetailsDiv = crashMap.StatsPane.getCrashDiv();
    crashDetailsDiv.appendChild(self.domNode);

    tableDiv.__ag_grid = this.grid ;
}
