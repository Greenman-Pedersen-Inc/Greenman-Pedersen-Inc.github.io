import ApplyChangesSelector from './widgets/ApplyChangesSelector';
import RadioBoxTable from './widgets/RadioBoxTable';
import SelectDropDown from './widgets/SelectDropDown';
import './multivehicle.filter.css';

/**
 * Returns the description for a code based on the categoryField
 * @date 2022-02-07
 * @param {string} targetCode - numeric code
 * @param {array} categoryFieldValues - array of [ {value: #, description: 'human-readable description'}, ... ]
 * @returns {string} 'human-readable description of a numeric code
 */
function findDescription(targetCode, categoryFieldValues) {
    for (let index = 0; index < categoryFieldValues.length; index++) {
        const categoryField = categoryFieldValues[index];
        if (categoryField.value === targetCode) {
            return categoryField.description;
        }
    }
}

/**
 * Creates the Multivehicle Filter in the filters menu.
 * @date 2024-11-20
 * @module FilterMenu/widgets/MultivehicleFilter
 * @param {domNode} parent The parent div of this filter toggle div.
 * @param {module} crashFilter The Crash Filter module.
 * @param {domNode} toggleAttachPoint The div where this filter toggle div will be appended to.
 * @param {domNode} filterContents The div where the filter options will be attached to.
 * @param {object} filterOptions The object containing the filter categories and their options.
 * @param {domNode} toggleButton The div where this filter toggle div will be appended to.
 * @param {module} crashMap The Crash Map module.
 * @param {module} resultsPane The Results Pane module.
 * @requires ./widgets/ApplyChangesSelector
 * @requires ./widgets/RadioBoxTable
 * @requires ./widgets/SelectDropDown
 * @requires ./multivehicle.filter.css
 */
export default function MultivehicleFilter(
    parent,
    crashFilter,
    toggleAttachPoint,
    filterContents,
    filterInfo,
    toggleButton,
    crashMap,
    resultsPane
) {
    const self = this;
    this.id = 'MultivehicleFilter';
    this.parentSidebar = toggleAttachPoint;
    this.filterMenuButton = toggleButton;

    /**
     * Clears the selected filters in this widget.
     * @date 2024-11-20
     * @function reset
     */
    this.reset = function () {
        vehicleOne.reset();
        vehicleTwo.reset();
        multivehicleSelector.reset();
    };

    /**
     * Gets the selected filter attributes and codes.
     * @date 2024-11-19
     * @function getSelections
     */
    this.getSelections = function () {
        if (Object.values(self.filter).filter((values) => values).length > 0) {
            self.domNode.classList.add('selected');
            self.toggle.classList.add('green-button-toggle');
            self.toggle.classList.remove('orange-button-toggle');
        } else {
            self.domNode.classList.remove('selected');
            self.toggle.classList.remove('green-button-toggle');
            if (self.domNode.classList.contains('hidden')) {
                self.toggle.classList.remove('orange-button-toggle');
            } else {
                self.toggle.classList.add('orange-button-toggle');
            }
        }

        return self.filter;
    };

    /**
     * Gets the human-readble descriptions based on the current selected filters.
     * @date 2024-11-19
     * @function getDescriptionSelections
     */
    this.getDescriptionSelections = function () {
        var filters = {};

        if (self.filter['tot_veh_involved']) {
            if (self.filter['tot_veh_involved'] !== undefined) {
                filters['Total Vehicles Involved'] = findDescription(
                    self.filter['tot_veh_involved'],
                    self.numberVehiclesFieldValues
                );
            }
        }

        if (self.filter['veh_one_travel_dir_code']) {
            if (self.filter['veh_one_travel_dir_code'] !== undefined) {
                filters['Vehicle 1 Travel Dir.'] = findDescription(
                    self.filter['veh_one_travel_dir_code'],
                    self.vehicleOneDirectionFieldValues
                );
            }
        }

        if (self.filter['veh_two_travel_dir_code']) {
            if (self.filter['veh_two_travel_dir_code'] !== undefined) {
                filters['Vehicle 2 Travel Dir.'] = findDescription(
                    self.filter['veh_two_travel_dir_code'],
                    self.vehicleTwoDirectionFieldValues
                );
            }
        }

        if (filters) {
            return filters;
        }
    };

    /**
     * Returns this widget based on the type `filterField`.
     * @date 2024-11-19
     * @function checkFilterField
     * @param {string} filterField The type of widget to access.
     * @returns {Module} This widget.
     */
    this.checkFilterField = function (filterField) {
        if (filterField === 'tot_veh_involved') {
            return this;
        }
    };

    this.searchFilters = function(searchText) {

    }


    /**
     * Sets the values for vehicle 1 and vehicle 2 in the Multivehicle filter.
     * @date 2024-11-20
     * @param {string} tot_veh_involved The filter value selected for the total number of vehicles involved.
     * @param {string} veh_one_travel_dir_code The filter value selected for the travel direction for vehicle 1.
     * @param {string} veh_two_travel_dir_code The filter value selected for the travel direction for vehicle 2.
     */
    this.setFilter = function (tot_veh_involved, veh_one_travel_dir_code, veh_two_travel_dir_code) {
        // self.filter['tot_veh_involved'] = tot_veh_involved;
        // self.filter['veh_one_travel_dir_code'] = veh_one_travel_dir_code;
        // self.filter['veh_two_travel_dir_code'] = veh_two_travel_dir_code;
        multivehicleSelector.update(tot_veh_involved);
        vehicleOne.update(veh_one_travel_dir_code);
        if (veh_two_travel_dir_code != '') vehicleTwo.update(veh_two_travel_dir_code);
    };

    this.domNode = document.createElement('div');

    /**
     * Hides the Multivehicle Filter widget.
     * @date 2024-12-17
     * @function hide
     */
    this.hide = function () {
        self.domNode.classList.add('hidden');
    };

    /**
     * Displays the Multivehicle Filter widget.
     * @date 2024-12-17
     * @function show
     */
    this.show = function () {
        self.domNode.classList.remove('hidden');
    };

    this.toggle = document.createElement('div');
    if (filterInfo.flyout.hasOwnProperty('tooltip')) {
        this.toggle.title = filterInfo.flyout.tooltip;
        if (filterInfo.label) {
            this.toggle.title += ` This filter corresponds to box number ${filterInfo.label.njtr1Number} in the NJTR-1 form.`
        } 
    }    this.toggle.className = 'omni-element';
    this.toggle.id = filterInfo.flyout.id + 'Toggle';
    this.toggle.innerHTML = `<span class="toggle-label">${filterInfo.flyout.title}</span>`;
    let icon = document.createElement('div');
    icon.className = `image-icon ${filterInfo.toggleIcon}`;

    this.toggle.prepend(icon);

    this.toggle.addEventListener('click', () => {
        const parentSidebar = self.parentSidebar;
        const filterContainers = document.querySelectorAll('.filter-container');
        const menuToggles = parentSidebar.querySelectorAll('.omni-element');

        if (self.domNode.classList.contains('hidden')) {
            filterContainers.forEach((filterContainer) => {
                filterContainer.classList.add('hidden');
            });

            menuToggles.forEach((toggle) => {
                if (toggle !== this.toggle) {
                    toggle.classList.remove('orange-button-toggle');
                }
            });

            Array.prototype.filter.call(parentSidebar.parentNode.children, function (child) {
                if (
                    child !== parentSidebar &&
                    child !== parent.content &&
                    child.classList.contains('sidebar-container')
                ) {
                    child.classList.add('hidden');
                }
            });
        }

        self.domNode.classList.toggle('hidden');
    });

    this.toggle.addEventListener('mouseenter', function (event) {
        if (!this.classList.contains('green-button-toggle')) {
            this.classList.add('orange-button-toggle');
        }
    });

    this.toggle.addEventListener('mouseleave', function (event) {
        if (self.domNode && self.domNode.classList.contains('hidden'))
            self.toggle.classList.remove('orange-button-toggle');
    });

    toggleAttachPoint.append(this.toggle);

    this.filter = {};

    const title = document.createElement('div');
    const content = document.createElement('div');
    const vehicleNumberSelectors = document.createElement('div');
    const widgetInfo = filterInfo.widgets.find((obj) => {
        return obj.type === 'multivehicle';
    });
    const numberVehiclesField = widgetInfo.fields['tot_veh_involved'];
    const vehicleOneDirectionField = widgetInfo.fields['veh_one_travel_dir_code'];
    const vehicleTwoDirectionField = widgetInfo.fields['veh_two_travel_dir_code'];
    const multivehicleSelectionDiv = document.createElement('div');
    const multivehicleSelector = new RadioBoxTable(self, 'tot_veh_involved', filterInfo, numberVehiclesField.values);
    const multivehicleDirectionDiv = document.createElement('div');
    const vehicleOne = new SelectDropDown(self, 'V1 ', 'veh_one_travel_dir_code', vehicleOneDirectionField.values);
    const vehicleTwo = new SelectDropDown(self, 'V2 ', 'veh_two_travel_dir_code', vehicleTwoDirectionField.values);
    const applyChangesSelector = new ApplyChangesSelector(self, crashMap, resultsPane);
    const applyRow = document.createElement('div');

    this.numberVehiclesFieldValues = numberVehiclesField.values;
    this.vehicleOneDirectionFieldValues = vehicleOneDirectionField.values;
    this.vehicleTwoDirectionFieldValues = vehicleTwoDirectionField.values;

    Object.defineProperty(this.filter, 'tot_veh_involved', {
        get: function () {
            return self['_' + 'tot_veh_involved'];
        },
        set: function (value) {
            if (value) {
                if (value.includes('2') || value.includes('>1')) {
                    vehicleOne.enable();
                    vehicleTwo.enable();
                } else if (value.includes('1') || value.includes('0')) {
                    vehicleOne.enable();
                    vehicleTwo.disable();
                } else {
                    vehicleOne.disable();
                    vehicleTwo.disable();
                }
                self['_' + 'tot_veh_involved'] = value;
                //multivehicleSelector['tot_veh_involved'] = value;
            } else {
                vehicleOne.disable();
                vehicleTwo.disable();
                self['_' + 'tot_veh_involved'] = undefined;
                //multivehicleSelector['tot_veh_involved'] = undefined;
            }
        },
        enumerable: true
    });

    Object.defineProperty(this.filter, 'veh_one_travel_dir_code', {
        get: function () {
            return self['_' + 'veh_one_travel_dir_code'];
        },
        set: function (value) {
            if (value === '') self['_' + 'veh_one_travel_dir_code'] = undefined;
            else self['_' + 'veh_one_travel_dir_code'] = value;
        },
        enumerable: true
    });

    Object.defineProperty(this.filter, 'veh_two_travel_dir_code', {
        get: function () {
            return self['_' + 'veh_two_travel_dir_code'];
        },
        set: function (value) {
            if (value === '') self['_' + 'veh_two_travel_dir_code'] = undefined;
            else self['_' + 'veh_two_travel_dir_code'] = value;
        },
        enumerable: true
    });

    // this is here because of some weird behavior with the define property function
    // sets the attribute to undefined.
    this.filter['tot_veh_involved'] = '';
    this.filter['veh_one_travel_dir_code'] = '';
    this.filter['veh_two_travel_dir_code'] = '';

    this.domNode.classList = 'filter-container omni-element has-title hidden';
    filterContents.appendChild(this.domNode);

    title.className = 'omni-title';
    title.innerHTML = 'Multivehicle Filter';
    this.domNode.appendChild(title);

    content.className = 'multivehicle-filter-root-container container';
    this.domNode.append(content);

    vehicleNumberSelectors.className = 'row';
    content.appendChild(vehicleNumberSelectors);

    multivehicleSelectionDiv.className = 'col';
    // multivehicleSelectionDiv.style = 'margin: 0.15em';
    vehicleNumberSelectors.appendChild(multivehicleSelectionDiv);

    multivehicleSelectionDiv.appendChild(multivehicleSelector.domNode);
    multivehicleDirectionDiv.className = 'col';
    // multivehicleDirectionDiv.style = 'margin: 0.3em';

    vehicleNumberSelectors.appendChild(multivehicleDirectionDiv);

    multivehicleDirectionDiv.appendChild(vehicleOne.domNode);
    multivehicleDirectionDiv.appendChild(vehicleTwo.domNode);

    applyRow.classList = 'row';
    // applyRow.style = 'font-size: small; margin: 0 0.2em 0.1em 0.2em; width: calc(100% - 0.4em); padding-top: 2px; cursor: pointer;';
    applyRow.appendChild(applyChangesSelector.domNode);

    content.appendChild(applyRow);
}
